32// Thin wrapper around preserveDebugSections to be used as a standalone plugin.
33class DebugInfoPreservationPlugin : public ObjectLinkingLayer::Plugin {
34public:
35 void modifyPassConfig(MaterializationResponsibility &MR,
36 jitlink::LinkGraph &LG,
37 jitlink::PassConfiguration &PassConfig) override {
38 PassConfig.PrePrunePasses.push_back(preserveDebugSections);
39 }
40
41 Error notifyRemovingResources(JITDylib &JD, ResourceKey K) override {
42 // Do nothing.
43 return Error::success();
44 }
45 Error notifyFailed(MaterializationResponsibility &MR) override {
46 // Do nothing.
47 return Error::success();
48 }
49 void notifyTransferringResources(JITDylib &JD, ResourceKey DstKey,
50 ResourceKey SrcKey) override {
51 // Do nothing.
52 }
53
54 static Expected<std::unique_ptr<DebugInfoPreservationPlugin>> Create() {
55 return std::make_unique<DebugInfoPreservationPlugin>();
56 }
57};
