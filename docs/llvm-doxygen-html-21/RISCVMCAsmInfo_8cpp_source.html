34const MCExpr *RISCVMCAsmInfo::getExprForFDESymbol(const MCSymbol *Sym,
35 unsigned Encoding,
36 MCStreamer &Streamer) const {
37 if (!(Encoding & dwarf::DW_EH_PE_pcrel))
38 return MCAsmInfo::getExprForFDESymbol(Sym, Encoding, Streamer);
39
40 // The default symbol subtraction results in an ADD/SUB relocation pair.
41 // Processing this relocation pair is problematic when linker relaxation is
42 // enabled, so we follow binutils in using the R_RISCV_32_PCREL relocation
43 // for the FDE initial location.
44 MCContext &Ctx = Streamer.getContext();
45 const MCExpr *ME = MCSymbolRefExpr::create(Sym, Ctx);
46 assert(Encoding & dwarf::DW_EH_PE_sdata4 && "Unexpected encoding");
47 return MCSpecifierExpr::create(ME, ELF::R_RISCV_32_PCREL, Ctx);
48}

50void RISCVMCAsmInfo::printSpecifierExpr(raw_ostream &OS,
51 const MCSpecifierExpr &Expr) const {
52 auto S = Expr.getSpecifier();
53 bool HasSpecifier = S != 0 && S != ELF::R_RISCV_CALL_PLT;
54 if (HasSpecifier)
55 OS << '%' << RISCV::getSpecifierName(S) << '(';
56 printExpr(OS, *Expr.getSubExpr());
57 if (HasSpecifier)
58 OS << ')';
59}
