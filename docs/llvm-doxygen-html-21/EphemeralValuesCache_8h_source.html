27/// A cache of ephemeral values within a function.
28class EphemeralValuesCache {
29 SmallPtrSet<const Value *, 32> EphValues;
30 Function &F;
31 AssumptionCache &AC;
32 bool Collected = false;
33
34 LLVM_ABI void collectEphemeralValues();
35
36public:
37 EphemeralValuesCache(Function &F, AssumptionCache &AC) : F(F), AC(AC) {}
38 void clear() {
39 EphValues.clear();
40 Collected = false;
41 }
42 const SmallPtrSetImpl<const Value *> &ephValues() {
43 if (!Collected)
44 collectEphemeralValues();
45 return EphValues;
46 }
47};
