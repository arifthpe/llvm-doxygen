68 auto &Subtarget = MF->getSubtarget<RISCVSubtarget>();

80 : CSR_XLEN_V_Interrupt_SaveList;
81 }
82 if (Subtarget.hasStdExtD())
83 return Subtarget.hasStdExtE() ? CSR_XLEN_F64_Interrupt_RVE_SaveList
84 : CSR_XLEN_F64_Interrupt_SaveList;
85 if (Subtarget.hasStdExtF())
86 return Subtarget.hasStdExtE() ? CSR_XLEN_F32_Interrupt_RVE_SaveList

89 : CSR_Interrupt_SaveList;
90 }
91
92 bool HasVectorCSR =
93 MF->getFunction().getCallingConv() == CallingConv::RISCV_VectorCall &&
94 Subtarget.hasVInstructions();
95
96 switch (Subtarget.getTargetABI()) {
97 default:
98 llvm_unreachable("Unrecognized ABI");
99 case RISCVABI::ABI_ILP32E:
100 case RISCVABI::ABI_LP64E:
101 return CSR_ILP32E_LP64E_SaveList;
102 case RISCVABI::ABI_ILP32:

106 return CSR_ILP32_LP64_SaveList;
107 case RISCVABI::ABI_ILP32F:
108 case RISCVABI::ABI_LP64F:
109 if (HasVectorCSR)
110 return CSR_ILP32F_LP64F_V_SaveList;
111 return CSR_ILP32F_LP64F_SaveList;

120BitVector RISCVRegisterInfo::getReservedRegs(const MachineFunction &MF) const {
121 const RISCVFrameLowering *TFI = getFrameLowering(MF);
122 BitVector Reserved(getNumRegs());
123 auto &Subtarget = MF.getSubtarget<RISCVSubtarget>();
124
125 for (size_t Reg = 0; Reg < getNumRegs(); Reg++) {
126 // Mark any GPRs requested to be reserved as such
127 if (Subtarget.isRegisterReservedByUser(Reg))
128 markSuperRegs(Reserved, Reg);
129
130 // Mark all the registers defined as constant in TableGen as reserved.
131 if (isConstantPhysReg(Reg))
132 markSuperRegs(Reserved, Reg);
133 }
134
135 // Use markSuperRegs to ensure any register aliases are also reserved
136 markSuperRegs(Reserved, RISCV::X2_H); // sp
137 markSuperRegs(Reserved, RISCV::X3_H); // gp
138 markSuperRegs(Reserved, RISCV::X4_H); // tp
139 if (TFI->hasFP(MF))
140 markSuperRegs(Reserved, RISCV::X8_H); // fp
141 // Reserve the base register if we need to realign the stack and allocate
142 // variable-sized objects at runtime.
143 if (TFI->hasBP(MF))
144 markSuperRegs(Reserved, RISCVABI::getBPReg()); // bp
145
146 // Additionally reserve dummy register used to form the register pair
147 // beginning with 'x0' for instructions that take register pairs.
148 markSuperRegs(Reserved, RISCV::DUMMY_REG_PAIR_WITH_X0);
149
150 // There are only 16 GPRs for RVE.
151 if (Subtarget.hasStdExtE())
152 for (MCPhysReg Reg = RISCV::X16_H; Reg <= RISCV::X31_H; Reg++)
153 markSuperRegs(Reserved, Reg);
154
155 // V registers for code generation. We handle them manually.
156 markSuperRegs(Reserved, RISCV::VL);
157 markSuperRegs(Reserved, RISCV::VTYPE);
158 markSuperRegs(Reserved, RISCV::VXSAT);
159 markSuperRegs(Reserved, RISCV::VXRM);
160
161 // Floating point environment registers.
162 markSuperRegs(Reserved, RISCV::FRM);
163 markSuperRegs(Reserved, RISCV::FFLAGS);
164
165 // SiFive VCIX state registers.
166 markSuperRegs(Reserved, RISCV::SF_VCIX_STATE);
167
168 if (MF.getFunction().getCallingConv() == CallingConv::GRAAL) {
169 if (Subtarget.hasStdExtE())
170 report_fatal_error("Graal reserved registers do not exist in RVE");
171 markSuperRegs(Reserved, RISCV::X23_H);
172 markSuperRegs(Reserved, RISCV::X27_H);
173 }
174
175 // Shadow stack pointer.
176 markSuperRegs(Reserved, RISCV::SSP);
177
178 assert(checkAllSuperRegsMarked(Reserved));
179 return Reserved;
180}

191void RISCVRegisterInfo::adjustReg(MachineBasicBlock &MBB,
192 MachineBasicBlock::iterator II,
193 const DebugLoc &DL, Register DestReg,
194 Register SrcReg, StackOffset Offset,
195 MachineInstr::MIFlag Flag,
196 MaybeAlign RequiredAlign) const {
197
198 if (DestReg == SrcReg && !Offset.getFixed() && !Offset.getScalable())
199 return;
200
201 MachineFunction &MF = *MBB.getParent();
202 MachineRegisterInfo &MRI = MF.getRegInfo();
203 const RISCVSubtarget &ST = MF.getSubtarget<RISCVSubtarget>();
204 const RISCVInstrInfo *TII = ST.getInstrInfo();
205
206 // Optimize compile time offset case
207 if (Offset.getScalable()) {
208 if (auto VLEN = ST.getRealVLen()) {
209 // 1. Multiply the number of v-slots by the (constant) length of register
210 const int64_t VLENB = *VLEN / 8;
211 assert(Offset.getScalable() % RISCV::RVVBytesPerBlock == 0 &&
212 "Reserve the stack by the multiple of one vector size.");
213 const int64_t NumOfVReg = Offset.getScalable() / 8;
214 const int64_t FixedOffset = NumOfVReg * VLENB;
215 if (!isInt<32>(FixedOffset)) {
216 report_fatal_error(
217 "Frame size outside of the signed 32-bit range not supported");
218 }
219 Offset = StackOffset::getFixed(FixedOffset + Offset.getFixed());
220 }
221 }
222
223 bool KillSrcReg = false;
224
225 if (Offset.getScalable()) {
226 unsigned ScalableAdjOpc = RISCV::ADD;
227 int64_t ScalableValue = Offset.getScalable();
228 if (ScalableValue < 0) {
229 ScalableValue = -ScalableValue;
230 ScalableAdjOpc = RISCV::SUB;
231 }
232 // Get vlenb and multiply vlen with the number of vector registers.
233 Register ScratchReg = DestReg;
234 if (DestReg == SrcReg)
235 ScratchReg = MRI.createVirtualRegister(&RISCV::GPRRegClass);
236
237 assert(ScalableValue > 0 && "There is no need to get VLEN scaled value.");
238 assert(ScalableValue % RISCV::RVVBytesPerBlock == 0 &&
239 "Reserve the stack by the multiple of one vector size.");
240 assert(isInt<32>(ScalableValue / RISCV::RVVBytesPerBlock) &&
241 "Expect the number of vector registers within 32-bits.");
242 uint32_t NumOfVReg = ScalableValue / RISCV::RVVBytesPerBlock;
243 // Only use vsetvli rather than vlenb if adjusting in the prologue or
244 // epilogue, otherwise it may disturb the VTYPE and VL status.
245 bool IsPrologueOrEpilogue =
246 Flag == MachineInstr::FrameSetup || Flag == MachineInstr::FrameDestroy;
247 bool UseVsetvliRatherThanVlenb =
248 IsPrologueOrEpilogue && ST.preferVsetvliOverReadVLENB();
249 if (UseVsetvliRatherThanVlenb && (NumOfVReg == 1 || NumOfVReg == 2 ||
250 NumOfVReg == 4 || NumOfVReg == 8)) {
251 BuildMI(MBB, II, DL, TII->get(RISCV::PseudoReadVLENBViaVSETVLIX0),
252 ScratchReg)
253 .addImm(NumOfVReg)
254 .setMIFlag(Flag);
255 BuildMI(MBB, II, DL, TII->get(ScalableAdjOpc), DestReg)
256 .addReg(SrcReg)
257 .addReg(ScratchReg, RegState::Kill)
258 .setMIFlag(Flag);
259 } else {
260 if (UseVsetvliRatherThanVlenb)
261 BuildMI(MBB, II, DL, TII->get(RISCV::PseudoReadVLENBViaVSETVLIX0),
262 ScratchReg)
263 .addImm(1)
264 .setMIFlag(Flag);
265 else
266 BuildMI(MBB, II, DL, TII->get(RISCV::PseudoReadVLENB), ScratchReg)
267 .setMIFlag(Flag);
268
269 if (ScalableAdjOpc == RISCV::ADD && ST.hasStdExtZba() &&
270 (NumOfVReg == 2 || NumOfVReg == 4 || NumOfVReg == 8)) {
271 unsigned Opc = NumOfVReg == 2
272 ? RISCV::SH1ADD
273 : (NumOfVReg == 4 ? RISCV::SH2ADD : RISCV::SH3ADD);
274 BuildMI(MBB, II, DL, TII->get(Opc), DestReg)
275 .addReg(ScratchReg, RegState::Kill)
276 .addReg(SrcReg)
277 .setMIFlag(Flag);
278 } else {
279 TII->mulImm(MF, MBB, II, DL, ScratchReg, NumOfVReg, Flag);
280 BuildMI(MBB, II, DL, TII->get(ScalableAdjOpc), DestReg)
281 .addReg(SrcReg)
282 .addReg(ScratchReg, RegState::Kill)
283 .setMIFlag(Flag);
284 }
285 }
286 SrcReg = DestReg;
287 KillSrcReg = true;
288 }
289
290 int64_t Val = Offset.getFixed();
291 if (DestReg == SrcReg && Val == 0)
292 return;
293
294 const uint64_t Align = RequiredAlign.valueOrOne().value();
295
296 if (isInt<12>(Val)) {
297 BuildMI(MBB, II, DL, TII->get(RISCV::ADDI), DestReg)
298 .addReg(SrcReg, getKillRegState(KillSrcReg))
299 .addImm(Val)
300 .setMIFlag(Flag);
301 return;
302 }
303
304 // Use the QC_E_ADDI instruction from the Xqcilia extension that can take a
305 // signed 26-bit immediate.
306 if (ST.hasVendorXqcilia() && isInt<26>(Val)) {
307 // The one case where using this instruction is sub-optimal is if Val can be
308 // materialized with a single compressible LUI and following add/sub is also
309 // compressible. Avoid doing this if that is the case.
310 int Hi20 = (Val & 0xFFFFF000) >> 12;
311 bool IsCompressLUI =
312 ((Val & 0xFFF) == 0) && (Hi20 != 0) &&
313 (isUInt<5>(Hi20) || (Hi20 >= 0xfffe0 && Hi20 <= 0xfffff));
314 bool IsCompressAddSub =
315 (SrcReg == DestReg) &&
316 ((Val > 0 && RISCV::GPRNoX0RegClass.contains(SrcReg)) ||
317 (Val < 0 && RISCV::GPRCRegClass.contains(SrcReg)));
318
319 if (!(IsCompressLUI && IsCompressAddSub)) {
320 BuildMI(MBB, II, DL, TII->get(RISCV::QC_E_ADDI), DestReg)
321 .addReg(SrcReg, getKillRegState(KillSrcReg))
322 .addImm(Val)
323 .setMIFlag(Flag);
324 return;
325 }
326 }
327
328 // Try to split the offset across two ADDIs. We need to keep the intermediate
329 // result aligned after each ADDI. We need to determine the maximum value we
330 // can put in each ADDI. In the negative direction, we can use -2048 which is
331 // always sufficiently aligned. In the positive direction, we need to find the
332 // largest 12-bit immediate that is aligned. Exclude -4096 since it can be
333 // created with LUI.
334 assert(Align < 2048 && "Required alignment too large");
335 int64_t MaxPosAdjStep = 2048 - Align;
336 if (Val > -4096 && Val <= (2 * MaxPosAdjStep)) {
337 int64_t FirstAdj = Val < 0 ? -2048 : MaxPosAdjStep;
338 Val -= FirstAdj;
339 BuildMI(MBB, II, DL, TII->get(RISCV::ADDI), DestReg)
340 .addReg(SrcReg, getKillRegState(KillSrcReg))
341 .addImm(FirstAdj)
342 .setMIFlag(Flag);
343 BuildMI(MBB, II, DL, TII->get(RISCV::ADDI), DestReg)
344 .addReg(DestReg, RegState::Kill)
345 .addImm(Val)
346 .setMIFlag(Flag);
347 return;
348 }
349
350 // Use shNadd if doing so lets us materialize a 12 bit immediate with a single
351 // instruction. This saves 1 instruction over the full lui/addi+add fallback
352 // path. We avoid anything which can be done with a single lui as it might
353 // be compressible. Note that the sh1add case is fully covered by the 2x addi
354 // case just above and is thus omitted.
355 if (ST.hasStdExtZba() && (Val & 0xFFF) != 0) {
356 unsigned Opc = 0;
357 if (isShiftedInt<12, 3>(Val)) {
358 Opc = RISCV::SH3ADD;
359 Val = Val >> 3;
360 } else if (isShiftedInt<12, 2>(Val)) {
361 Opc = RISCV::SH2ADD;
362 Val = Val >> 2;
363 }
364 if (Opc) {
365 Register ScratchReg = MRI.createVirtualRegister(&RISCV::GPRRegClass);
366 TII->movImm(MBB, II, DL, ScratchReg, Val, Flag);
367 BuildMI(MBB, II, DL, TII->get(Opc), DestReg)
368 .addReg(ScratchReg, RegState::Kill)
369 .addReg(SrcReg, getKillRegState(KillSrcReg))
370 .setMIFlag(Flag);
371 return;
372 }
373 }
374
375 unsigned Opc = RISCV::ADD;
376 if (Val < 0) {
377 Val = -Val;
378 Opc = RISCV::SUB;
379 }
380
381 Register ScratchReg = MRI.createVirtualRegister(&RISCV::GPRRegClass);
382 TII->movImm(MBB, II, DL, ScratchReg, Val, Flag);
383 BuildMI(MBB, II, DL, TII->get(Opc), DestReg)
384 .addReg(SrcReg, getKillRegState(KillSrcReg))
385 .addReg(ScratchReg, RegState::Kill)
386 .setMIFlag(Flag);
387}

389// Split a VSPILLx_Mx pseudo into multiple whole register stores separated by
390// LMUL*VLENB bytes.
391void RISCVRegisterInfo::lowerVSPILL(MachineBasicBlock::iterator II) const {
392 DebugLoc DL = II->getDebugLoc();
393 MachineBasicBlock &MBB = *II->getParent();
394 MachineFunction &MF = *MBB.getParent();
395 MachineRegisterInfo &MRI = MF.getRegInfo();
396 const RISCVSubtarget &STI = MF.getSubtarget<RISCVSubtarget>();
397 const TargetInstrInfo *TII = STI.getInstrInfo();
398 const TargetRegisterInfo *TRI = STI.getRegisterInfo();
399
400 auto ZvlssegInfo = RISCV::isRVVSpillForZvlsseg(II->getOpcode());
401 unsigned NF = ZvlssegInfo->first;
402 unsigned LMUL = ZvlssegInfo->second;
403 assert(NF * LMUL <= 8 && "Invalid NF/LMUL combinations.");
404 unsigned Opcode, SubRegIdx;
405 switch (LMUL) {
406 default:
407 llvm_unreachable("LMUL must be 1, 2, or 4.");
408 case 1:
409 Opcode = RISCV::VS1R_V;
410 SubRegIdx = RISCV::sub_vrm1_0;
411 break;
412 case 2:
413 Opcode = RISCV::VS2R_V;
414 SubRegIdx = RISCV::sub_vrm2_0;
415 break;
416 case 4:
417 Opcode = RISCV::VS4R_V;
418 SubRegIdx = RISCV::sub_vrm4_0;
419 break;
420 }
421 static_assert(RISCV::sub_vrm1_7 == RISCV::sub_vrm1_0 + 7,
422 "Unexpected subreg numbering");
423 static_assert(RISCV::sub_vrm2_3 == RISCV::sub_vrm2_0 + 3,
424 "Unexpected subreg numbering");
425 static_assert(RISCV::sub_vrm4_1 == RISCV::sub_vrm4_0 + 1,
426 "Unexpected subreg numbering");
427
428 Register VL = MRI.createVirtualRegister(&RISCV::GPRRegClass);
429 // Optimize for constant VLEN.
430 if (auto VLEN = STI.getRealVLen()) {
431 const int64_t VLENB = *VLEN / 8;
432 int64_t Offset = VLENB * LMUL;
433 STI.getInstrInfo()->movImm(MBB, II, DL, VL, Offset);
434 } else {
435 BuildMI(MBB, II, DL, TII->get(RISCV::PseudoReadVLENB), VL);
436 uint32_t ShiftAmount = Log2_32(LMUL);
437 if (ShiftAmount != 0)
438 BuildMI(MBB, II, DL, TII->get(RISCV::SLLI), VL)
439 .addReg(VL)
440 .addImm(ShiftAmount);
441 }
442
443 Register SrcReg = II->getOperand(0).getReg();
444 Register Base = II->getOperand(1).getReg();
445 bool IsBaseKill = II->getOperand(1).isKill();
446 Register NewBase = MRI.createVirtualRegister(&RISCV::GPRRegClass);
447
448 auto *OldMMO = *(II->memoperands_begin());
449 LocationSize OldLoc = OldMMO->getSize();
450 assert(OldLoc.isPrecise() && OldLoc.getValue().isKnownMultipleOf(NF));
451 TypeSize NewSize = OldLoc.getValue().divideCoefficientBy(NF);
452 auto *NewMMO = MF.getMachineMemOperand(OldMMO, OldMMO->getOffset(), NewSize);
453 for (unsigned I = 0; I < NF; ++I) {
454 // Adding implicit-use of super register to describe we are using part of
455 // super register, that prevents machine verifier complaining when part of
456 // subreg is undef, see comment in MachineVerifier::checkLiveness for more
457 // detail.
458 BuildMI(MBB, II, DL, TII->get(Opcode))
459 .addReg(TRI->getSubReg(SrcReg, SubRegIdx + I))
460 .addReg(Base, getKillRegState(I == NF - 1))
461 .addMemOperand(NewMMO)
462 .addReg(SrcReg, RegState::Implicit);
463 if (I != NF - 1)
464 BuildMI(MBB, II, DL, TII->get(RISCV::ADD), NewBase)
465 .addReg(Base, getKillRegState(I != 0 || IsBaseKill))
466 .addReg(VL, getKillRegState(I == NF - 2));
467 Base = NewBase;
468 }
469 II->eraseFromParent();
470}

474void RISCVRegisterInfo::lowerVRELOAD(MachineBasicBlock::iterator II) const {
475 DebugLoc DL = II->getDebugLoc();

481 const TargetRegisterInfo *TRI = STI.getRegisterInfo();
482
483 auto ZvlssegInfo = RISCV::isRVVSpillForZvlsseg(II->getOpcode());
484 unsigned NF = ZvlssegInfo->first;
485 unsigned LMUL = ZvlssegInfo->second;
486 assert(NF * LMUL <= 8 && "Invalid NF/LMUL combinations.");
487 unsigned Opcode, SubRegIdx;
488 switch (LMUL) {
489 default:
490 llvm_unreachable("LMUL must be 1, 2, or 4.");
491 case 1:
492 Opcode = RISCV::VL1RE8_V;
493 SubRegIdx = RISCV::sub_vrm1_0;
494 break;
495 case 2:
496 Opcode = RISCV::VL2RE8_V;
497 SubRegIdx = RISCV::sub_vrm2_0;
498 break;
499 case 4:
500 Opcode = RISCV::VL4RE8_V;
501 SubRegIdx = RISCV::sub_vrm4_0;
502 break;
503 }
504 static_assert(RISCV::sub_vrm1_7 == RISCV::sub_vrm1_0 + 7,
505 "Unexpected subreg numbering");
506 static_assert(RISCV::sub_vrm2_3 == RISCV::sub_vrm2_0 + 3,
507 "Unexpected subreg numbering");
508 static_assert(RISCV::sub_vrm4_1 == RISCV::sub_vrm4_0 + 1,
509 "Unexpected subreg numbering");
510
511 Register VL = MRI.createVirtualRegister(&RISCV::GPRRegClass);
512 // Optimize for constant VLEN.
513 if (auto VLEN = STI.getRealVLen()) {
514 const int64_t VLENB = *VLEN / 8;
515 int64_t Offset = VLENB * LMUL;
516 STI.getInstrInfo()->movImm(MBB, II, DL, VL, Offset);
517 } else {
518 BuildMI(MBB, II, DL, TII->get(RISCV::PseudoReadVLENB), VL);
519 uint32_t ShiftAmount = Log2_32(LMUL);
520 if (ShiftAmount != 0)
521 BuildMI(MBB, II, DL, TII->get(RISCV::SLLI), VL)
522 .addReg(VL)
523 .addImm(ShiftAmount);
524 }
525
526 Register DestReg = II->getOperand(0).getReg();
527 Register Base = II->getOperand(1).getReg();
528 bool IsBaseKill = II->getOperand(1).isKill();
529 Register NewBase = MRI.createVirtualRegister(&RISCV::GPRRegClass);
530 auto *OldMMO = *(II->memoperands_begin());
531 LocationSize OldLoc = OldMMO->getSize();
532 assert(OldLoc.isPrecise() && OldLoc.getValue().isKnownMultipleOf(NF));
533 TypeSize NewSize = OldLoc.getValue().divideCoefficientBy(NF);
534 auto *NewMMO = MF.getMachineMemOperand(OldMMO, OldMMO->getOffset(), NewSize);
535 for (unsigned I = 0; I < NF; ++I) {
536 BuildMI(MBB, II, DL, TII->get(Opcode),
537 TRI->getSubReg(DestReg, SubRegIdx + I))
538 .addReg(Base, getKillRegState(I == NF - 1))

549bool RISCVRegisterInfo::eliminateFrameIndex(MachineBasicBlock::iterator II,
550 int SPAdj, unsigned FIOperandNum,
551 RegScavenger *RS) const {
552 assert(SPAdj == 0 && "Unexpected non-zero SPAdj value");
553
554 MachineInstr &MI = *II;
555 MachineFunction &MF = *MI.getParent()->getParent();
556 MachineRegisterInfo &MRI = MF.getRegInfo();
557 DebugLoc DL = MI.getDebugLoc();
558
559 int FrameIndex = MI.getOperand(FIOperandNum).getIndex();
560 Register FrameReg;
561 StackOffset Offset =
562 getFrameLowering(MF)->getFrameIndexReference(MF, FrameIndex, FrameReg);
563 bool IsRVVSpill = RISCV::isRVVSpill(MI);
564 if (!IsRVVSpill)
565 Offset += StackOffset::getFixed(MI.getOperand(FIOperandNum + 1).getImm());
566
567 if (!isInt<32>(Offset.getFixed())) {
568 report_fatal_error(
569 "Frame offsets outside of the signed 32-bit range not supported");
570 }
571
572 if (!IsRVVSpill) {
573 int64_t Val = Offset.getFixed();
574 int64_t Lo12 = SignExtend64<12>(Val);
575 unsigned Opc = MI.getOpcode();
576 if (Opc == RISCV::ADDI && !isInt<12>(Val)) {
577 // We chose to emit the canonical immediate sequence rather than folding
578 // the offset into the using add under the theory that doing so doesn't
579 // save dynamic instruction count and some target may fuse the canonical
580 // 32 bit immediate sequence. We still need to clear the portion of the
581 // offset encoded in the immediate.
582 MI.getOperand(FIOperandNum + 1).ChangeToImmediate(0);
583 } else if ((Opc == RISCV::PREFETCH_I || Opc == RISCV::PREFETCH_R ||
584 Opc == RISCV::PREFETCH_W) &&
585 (Lo12 & 0b11111) != 0) {
586 // Prefetch instructions require the offset to be 32 byte aligned.
587 MI.getOperand(FIOperandNum + 1).ChangeToImmediate(0);
588 } else if ((Opc == RISCV::PseudoRV32ZdinxLD ||
589 Opc == RISCV::PseudoRV32ZdinxSD) &&
590 Lo12 >= 2044) {
591 // This instruction will be split into 2 instructions. The second
592 // instruction will add 4 to the immediate. If that would overflow 12
593 // bits, we can't fold the offset.
594 MI.getOperand(FIOperandNum + 1).ChangeToImmediate(0);
595 } else {
596 // We can encode an add with 12 bit signed immediate in the immediate
597 // operand of our user instruction. As a result, the remaining
598 // offset can by construction, at worst, a LUI and a ADD.
599 MI.getOperand(FIOperandNum + 1).ChangeToImmediate(Lo12);
600 Offset = StackOffset::get((uint64_t)Val - (uint64_t)Lo12,
601 Offset.getScalable());
602 }
603 }
604
605 if (Offset.getScalable() || Offset.getFixed()) {
606 Register DestReg;
607 if (MI.getOpcode() == RISCV::ADDI)
608 DestReg = MI.getOperand(0).getReg();
609 else
610 DestReg = MRI.createVirtualRegister(&RISCV::GPRRegClass);
611 adjustReg(*II->getParent(), II, DL, DestReg, FrameReg, Offset,
612 MachineInstr::NoFlags, std::nullopt);
613 MI.getOperand(FIOperandNum).ChangeToRegister(DestReg, /*IsDef*/false,
614 /*IsImp*/false,
615 /*IsKill*/true);
616 } else {
617 MI.getOperand(FIOperandNum).ChangeToRegister(FrameReg, /*IsDef*/false,
618 /*IsImp*/false,
619 /*IsKill*/false);
620 }
621
622 // If after materializing the adjustment, we have a pointless ADDI, remove it
623 if (MI.getOpcode() == RISCV::ADDI &&
624 MI.getOperand(0).getReg() == MI.getOperand(1).getReg() &&
625 MI.getOperand(2).getImm() == 0) {
626 MI.eraseFromParent();
627 return true;
628 }

630 // Handle spill/fill of synthetic register classes for segment operations to
631 // ensure correctness in the edge case one gets spilled. There are many
632 // possible optimizations here, but given the extreme rarity of such spills,
633 // we prefer simplicity of implementation for now.
634 switch (MI.getOpcode()) {
635 case RISCV::PseudoVSPILL2_M1:
636 case RISCV::PseudoVSPILL2_M2:
637 case RISCV::PseudoVSPILL2_M4:
638 case RISCV::PseudoVSPILL3_M1:
639 case RISCV::PseudoVSPILL3_M2:
640 case RISCV::PseudoVSPILL4_M1:
641 case RISCV::PseudoVSPILL4_M2:
642 case RISCV::PseudoVSPILL5_M1:
643 case RISCV::PseudoVSPILL6_M1:
644 case RISCV::PseudoVSPILL7_M1:
645 case RISCV::PseudoVSPILL8_M1:
646 lowerVSPILL(II);
647 return true;
648 case RISCV::PseudoVRELOAD2_M1:
649 case RISCV::PseudoVRELOAD2_M2:
650 case RISCV::PseudoVRELOAD2_M4:
651 case RISCV::PseudoVRELOAD3_M1:
652 case RISCV::PseudoVRELOAD3_M2:
653 case RISCV::PseudoVRELOAD4_M1:
654 case RISCV::PseudoVRELOAD4_M2:
655 case RISCV::PseudoVRELOAD5_M1:
656 case RISCV::PseudoVRELOAD6_M1:
657 case RISCV::PseudoVRELOAD7_M1:
658 case RISCV::PseudoVRELOAD8_M1:

671// Returns true if the instruction's frame index reference would be better
672// served by a base register other than FP or SP.
673// Used by LocalStackSlotAllocation pass to determine which frame index
674// references it should create new base registers for.
675bool RISCVRegisterInfo::needsFrameBaseReg(MachineInstr *MI,
676 int64_t Offset) const {
677 unsigned FIOperandNum = 0;
678 for (; !MI->getOperand(FIOperandNum).isFI(); FIOperandNum++)
679 assert(FIOperandNum < MI->getNumOperands() &&
680 "Instr doesn't have FrameIndex operand");
681
682 // For RISC-V, The machine instructions that include a FrameIndex operand
683 // are load/store, ADDI instructions.
684 unsigned MIFrm = RISCVII::getFormat(MI->getDesc().TSFlags);
685 if (MIFrm != RISCVII::InstFormatI && MIFrm != RISCVII::InstFormatS)
686 return false;
687 // We only generate virtual base registers for loads and stores, so
688 // return false for everything else.
689 if (!MI->mayLoad() && !MI->mayStore())
690 return false;
691
692 const MachineFunction &MF = *MI->getMF();
693 const MachineFrameInfo &MFI = MF.getFrameInfo();
694 const RISCVFrameLowering *TFI = getFrameLowering(MF);
695 const MachineRegisterInfo &MRI = MF.getRegInfo();
696
697 if (TFI->hasFP(MF) && !shouldRealignStack(MF)) {
698 auto &Subtarget = MF.getSubtarget<RISCVSubtarget>();
699 // Estimate the stack size used to store callee saved registers(
700 // excludes reserved registers).
701 unsigned CalleeSavedSize = 0;
702 for (const MCPhysReg *R = MRI.getCalleeSavedRegs(); MCPhysReg Reg = *R;
703 ++R) {
704 if (Subtarget.isRegisterReservedByUser(Reg))
705 continue;
706
707 if (RISCV::GPRRegClass.contains(Reg))
708 CalleeSavedSize += getSpillSize(RISCV::GPRRegClass);
709 else if (RISCV::FPR64RegClass.contains(Reg))
710 CalleeSavedSize += getSpillSize(RISCV::FPR64RegClass);
711 else if (RISCV::FPR32RegClass.contains(Reg))
712 CalleeSavedSize += getSpillSize(RISCV::FPR32RegClass);
713 // Ignore vector registers.
714 }
715
716 int64_t MaxFPOffset = Offset - CalleeSavedSize;
717 return !isFrameOffsetLegal(MI, RISCV::X8, MaxFPOffset);
718 }
719
720 // Assume 128 bytes spill slots size to estimate the maximum possible
721 // offset relative to the stack pointer.
722 // FIXME: The 128 is copied from ARM. We should run some statistics and pick a
723 // real one for RISC-V.
724 int64_t MaxSPOffset = Offset + 128;
725 MaxSPOffset += MFI.getLocalFrameSize();
726 return !isFrameOffsetLegal(MI, RISCV::X2, MaxSPOffset);
727}

893// Add two address hints to improve chances of being able to use a compressed
894// instruction.
895bool RISCVRegisterInfo::getRegAllocationHints(
896 Register VirtReg, ArrayRef<MCPhysReg> Order,
897 SmallVectorImpl<MCPhysReg> &Hints, const MachineFunction &MF,
898 const VirtRegMap *VRM, const LiveRegMatrix *Matrix) const {
899 const MachineRegisterInfo *MRI = &MF.getRegInfo();
900 auto &Subtarget = MF.getSubtarget<RISCVSubtarget>();
901
902 bool BaseImplRetVal = TargetRegisterInfo::getRegAllocationHints(
903 VirtReg, Order, Hints, MF, VRM, Matrix);
904
905 if (!VRM || DisableRegAllocHints)
906 return BaseImplRetVal;
907
908 // Add any two address hints after any copy hints.
909 SmallSet<Register, 4> TwoAddrHints;
910
911 auto tryAddHint = [&](const MachineOperand &VRRegMO, const MachineOperand &MO,
912 bool NeedGPRC) -> void {
913 Register Reg = MO.getReg();
914 Register PhysReg = Reg.isPhysical() ? Reg : Register(VRM->getPhys(Reg));
915 // TODO: Support GPRPair subregisters? Need to be careful with even/odd
916 // registers. If the virtual register is an odd register of a pair and the
917 // physical register is even (or vice versa), we should not add the hint.
918 if (PhysReg && (!NeedGPRC || RISCV::GPRCRegClass.contains(PhysReg)) &&
919 !MO.getSubReg() && !VRRegMO.getSubReg()) {
920 if (!MRI->isReserved(PhysReg) && !is_contained(Hints, PhysReg))
921 TwoAddrHints.insert(PhysReg);
922 }
923 };
924
925 // This is all of the compressible binary instructions. If an instruction
926 // needs GPRC register class operands \p NeedGPRC will be set to true.
927 auto isCompressible = [&Subtarget](const MachineInstr &MI, bool &NeedGPRC) {
928 NeedGPRC = false;
929 switch (MI.getOpcode()) {
930 default:
931 return false;
932 case RISCV::AND:
933 case RISCV::OR:
934 case RISCV::XOR:
935 case RISCV::SUB:
936 case RISCV::ADDW:
937 case RISCV::SUBW:
938 NeedGPRC = true;
939 return true;
940 case RISCV::ANDI: {
941 NeedGPRC = true;
942 if (!MI.getOperand(2).isImm())
943 return false;
944 int64_t Imm = MI.getOperand(2).getImm();
945 if (isInt<6>(Imm))
946 return true;
947 // c.zext.b
948 return Subtarget.hasStdExtZcb() && Imm == 255;
949 }
950 case RISCV::SRAI:
951 case RISCV::SRLI:
952 NeedGPRC = true;
953 return true;
954 case RISCV::ADD:
955 case RISCV::SLLI:
956 return true;
957 case RISCV::ADDI:
958 case RISCV::ADDIW:
959 return MI.getOperand(2).isImm() && isInt<6>(MI.getOperand(2).getImm());
960 case RISCV::MUL:
961 case RISCV::SEXT_B:
962 case RISCV::SEXT_H:
963 case RISCV::ZEXT_H_RV32:
964 case RISCV::ZEXT_H_RV64:
965 // c.mul, c.sext.b, c.sext.h, c.zext.h
966 NeedGPRC = true;
967 return Subtarget.hasStdExtZcb();
968 case RISCV::ADD_UW:
969 // c.zext.w
970 NeedGPRC = true;
971 return Subtarget.hasStdExtZcb() && MI.getOperand(2).isReg() &&
972 MI.getOperand(2).getReg() == RISCV::X0;
973 case RISCV::XORI:
974 // c.not
975 NeedGPRC = true;
976 return Subtarget.hasStdExtZcb() && MI.getOperand(2).isImm() &&
977 MI.getOperand(2).getImm() == -1;
978 }
979 };
980
981 // Returns true if this operand is compressible. For non-registers it always
982 // returns true. Immediate range was already checked in isCompressible.
983 // For registers, it checks if the register is a GPRC register. reg-reg
984 // instructions that require GPRC need all register operands to be GPRC.
985 auto isCompressibleOpnd = [&](const MachineOperand &MO) {
986 if (!MO.isReg())
987 return true;
988 Register Reg = MO.getReg();
989 Register PhysReg = Reg.isPhysical() ? Reg : Register(VRM->getPhys(Reg));
990 return PhysReg && RISCV::GPRCRegClass.contains(PhysReg);
991 };
992
993 for (auto &MO : MRI->reg_nodbg_operands(VirtReg)) {
994 const MachineInstr &MI = *MO.getParent();
995 unsigned OpIdx = MO.getOperandNo();
996 bool NeedGPRC;
997 if (isCompressible(MI, NeedGPRC)) {
998 if (OpIdx == 0 && MI.getOperand(1).isReg()) {
999 if (!NeedGPRC || MI.getNumExplicitOperands() < 3 ||
1000 MI.getOpcode() == RISCV::ADD_UW ||
1001 isCompressibleOpnd(MI.getOperand(2)))
1002 tryAddHint(MO, MI.getOperand(1), NeedGPRC);
1003 if (MI.isCommutable() && MI.getOperand(2).isReg() &&
1004 (!NeedGPRC || isCompressibleOpnd(MI.getOperand(1))))
1005 tryAddHint(MO, MI.getOperand(2), NeedGPRC);
1006 } else if (OpIdx == 1 && (!NeedGPRC || MI.getNumExplicitOperands() < 3 ||
1007 isCompressibleOpnd(MI.getOperand(2)))) {
1008 tryAddHint(MO, MI.getOperand(0), NeedGPRC);
1009 } else if (MI.isCommutable() && OpIdx == 2 &&
1010 (!NeedGPRC || isCompressibleOpnd(MI.getOperand(1)))) {
1011 tryAddHint(MO, MI.getOperand(0), NeedGPRC);
1012 }
1013 }
1014
1015 // Add a hint if it would allow auipc/lui+addi(w) fusion.
1016 if ((MI.getOpcode() == RISCV::ADDIW || MI.getOpcode() == RISCV::ADDI) &&
1017 MI.getOperand(1).isReg()) {
1018 const MachineBasicBlock &MBB = *MI.getParent();
1019 MachineBasicBlock::const_iterator I = MI.getIterator();
1020 // Is the previous instruction a LUI or AUIPC that can be fused?
1021 if (I != MBB.begin()) {
1022 I = skipDebugInstructionsBackward(std::prev(I), MBB.begin());
1023 if (((I->getOpcode() == RISCV::LUI && Subtarget.hasLUIADDIFusion()) ||
1024 (I->getOpcode() == RISCV::AUIPC &&
1025 Subtarget.hasAUIPCADDIFusion())) &&
1026 I->getOperand(0).getReg() == MI.getOperand(1).getReg()) {
1027 if (OpIdx == 0)
1028 tryAddHint(MO, MI.getOperand(1), /*NeedGPRC=*/false);
1029 else
1030 tryAddHint(MO, MI.getOperand(0), /*NeedGPRC=*/false);
1031 }
1032 }
1033 }
1034 }
1035
1036 for (MCPhysReg OrderReg : Order)
1037 if (TwoAddrHints.count(OrderReg))
1038 Hints.push_back(OrderReg);
1039
1040 return BaseImplRetVal;
1041}
