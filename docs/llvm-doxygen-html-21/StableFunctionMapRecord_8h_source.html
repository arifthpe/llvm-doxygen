55 LLVM_ABI void deserializeYAML(yaml::Input &YIS);
56
57 /// Finalize the stable function map by trimming content.
58 void finalize(bool SkipTrim = false) { FunctionMap->finalize(SkipTrim); }
59
60 /// Merge the stable function map into this one.
61 void merge(const StableFunctionMapRecord &Other) {
62 FunctionMap->merge(*Other.FunctionMap);
63 }
64
65 /// \returns true if the stable function map is empty.
66 bool empty() const { return FunctionMap->empty(); }
67
68 /// Print the stable function map in a YAML format.

70 yaml::Output YOS(OS);
