41bool isELFInitializerSection(StringRef SecName) {
42 for (StringRef InitSection : ELFInitSectionNames) {
43 StringRef Name = SecName;
44 if (Name.consume_front(InitSection) && (Name.empty() || Name[0] == '.'))
