58raw_ostream &llvm::operator<<(raw_ostream &OS, CaptureComponents CC) {
59 if (capturesNothing(CC)) {
60 OS << "none";
61 return OS;
62 }
63
64 ListSeparator LS;
65 if (capturesAddressIsNullOnly(CC))
66 OS << LS << "address_is_null";
67 else if (capturesAddress(CC))
68 OS << LS << "address";
69 if (capturesReadProvenanceOnly(CC))
70 OS << LS << "read_provenance";
71 if (capturesFullProvenance(CC))
72 OS << LS << "provenance";
73
74 return OS;
75}

77raw_ostream &llvm::operator<<(raw_ostream &OS, CaptureInfo CI) {
78 ListSeparator LS;
79 CaptureComponents Other = CI.getOtherComponents();
80 CaptureComponents Ret = CI.getRetComponents();
81
82 OS << "captures(";
83 if (!capturesNothing(Other) || Other == Ret)
84 OS << LS << Other;
85 if (Other != Ret)
86 OS << LS << "ret: " << Ret;
87 OS << ")";
88 return OS;
