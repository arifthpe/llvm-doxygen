28// A bit mixer with very low latency using one multiplications and one
29// xor-shift. The constant is from splitmix64.
30inline uint64_t mix(uint64_t x) {
31 x *= 0xbf58476d1ce4e5b9u;
32 x ^= x >> 31;
33 return x;
34}

39/// Simplistic combination of 32-bit hash values into 32-bit hash values.
40inline unsigned combineHashValue(unsigned a, unsigned b) {
41 uint64_t x = (uint64_t)a << 32 | (uint64_t)b;
42 return (unsigned)densemap::detail::mix(x);
43}

60// Provide DenseMapInfo for all pointers. Come up with sentinel pointer values
61// that are aligned to alignof(T) bytes, but try to avoid requiring T to be
62// complete. This allows clients to instantiate DenseMap<T*, ...> with forward
63// declared key types. Assume that no pointer key type requires more than 4096
64// bytes of alignment.
65template<typename T>
66struct DenseMapInfo<T*> {
67 // The following should hold, but it would require T to be complete:
68 // static_assert(alignof(T) <= (1 << Log2MaxAlign),
69 // "DenseMap does not support pointer keys requiring more than "
70 // "Log2MaxAlign bits of alignment");
71 static constexpr uintptr_t Log2MaxAlign = 12;
72
73 static inline T* getEmptyKey() {
74 uintptr_t Val = static_cast<uintptr_t>(-1);
75 Val <<= Log2MaxAlign;
76 return reinterpret_cast<T*>(Val);
77 }
78
79 static inline T* getTombstoneKey() {
80 uintptr_t Val = static_cast<uintptr_t>(-2);
81 Val <<= Log2MaxAlign;
82 return reinterpret_cast<T*>(Val);
83 }
84
85 static unsigned getHashValue(const T *PtrVal) {
86 return (unsigned((uintptr_t)PtrVal) >> 4) ^
87 (unsigned((uintptr_t)PtrVal) >> 9);
88 }
89
90 static bool isEqual(const T *LHS, const T *RHS) { return LHS == RHS; }
91};

220// Provide DenseMapInfo for all pairs whose members have info.
221template<typename T, typename U>
222struct DenseMapInfo<std::pair<T, U>> {
223 using Pair = std::pair<T, U>;
224 using FirstInfo = DenseMapInfo<T>;
225 using SecondInfo = DenseMapInfo<U>;
226
227 static inline Pair getEmptyKey() {
228 return std::make_pair(FirstInfo::getEmptyKey(),
229 SecondInfo::getEmptyKey());
230 }
231
232 static inline Pair getTombstoneKey() {
233 return std::make_pair(FirstInfo::getTombstoneKey(),
234 SecondInfo::getTombstoneKey());
235 }
236
237 static unsigned getHashValue(const Pair& PairVal) {
238 return detail::combineHashValue(FirstInfo::getHashValue(PairVal.first),
239 SecondInfo::getHashValue(PairVal.second));
240 }
241
242 // Expose an additional function intended to be used by other
243 // specializations of DenseMapInfo without needing to know how
244 // to combine hash values manually
245 static unsigned getHashValuePiecewise(const T &First, const U &Second) {
246 return detail::combineHashValue(FirstInfo::getHashValue(First),
247 SecondInfo::getHashValue(Second));
248 }
249
250 static bool isEqual(const Pair &LHS, const Pair &RHS) {
251 return FirstInfo::isEqual(LHS.first, RHS.first) &&
252 SecondInfo::isEqual(LHS.second, RHS.second);
253 }
254};

268 template <unsigned I>
269 static unsigned getHashValueImpl(const Tuple &values, std::false_type) {
270 using EltType = std::tuple_element_t<I, Tuple>;
271 std::integral_constant<bool, I + 1 == sizeof...(Ts)> atEnd;
272 return detail::combineHashValue(
273 DenseMapInfo<EltType>::getHashValue(std::get<I>(values)),
274 getHashValueImpl<I + 1>(values, atEnd));
275 }
276
277 template <unsigned I>
278 static unsigned getHashValueImpl(const Tuple &, std::true_type) {
279 return 0;
280 }
281
282 static unsigned getHashValue(const std::tuple<Ts...> &values) {
283 std::integral_constant<bool, 0 == sizeof...(Ts)> atEnd;
284 return getHashValueImpl<0>(values, atEnd);
285 }

287 template <unsigned I>
288 static bool isEqualImpl(const Tuple &lhs, const Tuple &rhs, std::false_type) {
289 using EltType = std::tuple_element_t<I, Tuple>;
290 std::integral_constant<bool, I + 1 == sizeof...(Ts)> atEnd;
291 return DenseMapInfo<EltType>::isEqual(std::get<I>(lhs), std::get<I>(rhs)) &&

293 }

297 return true;
298 }
299

301 std::integral_constant<bool, 0 == sizeof...(Ts)> atEnd;

325template <typename T> struct DenseMapInfo<std::optional<T>> {
326 using Optional = std::optional<T>;
