55/// A simple container for information about the supported runtime calls.
56struct RuntimeLibcallsInfo {
57 explicit RuntimeLibcallsInfo(
58 const Triple &TT,
59 ExceptionHandling ExceptionModel = ExceptionHandling::None,
60 FloatABI::ABIType FloatABI = FloatABI::Default,
61 EABI EABIVersion = EABI::Default, StringRef ABIName = "") {
62 // FIXME: The ExceptionModel parameter is to handle the field in
63 // TargetOptions. This interface fails to distinguish the forced disable
64 // case for targets which support exceptions by default. This should
65 // probably be a module flag and removed from TargetOptions.
66 if (ExceptionModel == ExceptionHandling::None)
67 ExceptionModel = TT.getDefaultExceptionHandling();
68
69 initLibcalls(TT, ExceptionModel, FloatABI, EABIVersion, ABIName);
70 }

115 /// Return a function name compatible with RTLIB::MEMCPY, or nullptr if fully

152 static bool darwinHasSinCosStret(const Triple &TT) {
153 if (!TT.isOSDarwin())
154 return false;
155
156 // Don't bother with 32 bit x86.
157 if (TT.getArch() == Triple::x86)
158 return false;
159 // Macos < 10.9 has no sincos_stret.
160 if (TT.isMacOSX())
161 return !TT.isMacOSXVersionLT(10, 9) && TT.isArch64Bit();
162 // iOS < 7.0 has no sincos_stret.
163 if (TT.isiOS())
164 return !TT.isOSVersionLT(7, 0);
165 // Any other darwin such as WatchOS/TvOS is new enough.
166 return true;
167 }
168
169 static bool darwinHasExp10(const Triple &TT);
170
171 /// Return true if the target has sincosf/sincos/sincosl functions
172 static bool hasSinCos(const Triple &TT) {
173 return TT.isGNUEnvironment() || TT.isOSFuchsia() ||
174 (TT.isAndroid() && !TT.isAndroidVersionLT(9));
175 }
176
177 static bool hasSinCos_f32_f64(const Triple &TT) {
178 return hasSinCos(TT) || TT.isPS();
179 }
