13unsigned VecUtils::getFloorPowerOf2(unsigned Num) {
14 if (Num == 0)
15 return Num;
16 unsigned Mask = Num;
17 Mask >>= 1;
18 for (unsigned ShiftBy = 1; ShiftBy < sizeof(Num) * 8; ShiftBy <<= 1)
19 Mask |= Mask >> ShiftBy;
20 return Num & ~Mask;
21}
