40 // The type of this section.

43 // This section is a BSS section.

46 // Indicates that the PR symbol needs to set the length of the section to a
47 // non-zero value. This is only a problem with the ADA PR - the binder will

51 // Set to true if the section definition was already emitted.

57 MCSectionGOFF(StringRef Name, SectionKind K, bool IsVirtual,

94 bool isSD() const { return SymbolType == GOFF::ESD_ST_SectionDefinition; }

97
98 // Accessors to the attributes.
99 GOFF::SDAttr getSDAttributes() const {
100 assert(isSD() && "Not a SD section");
101 return SDAttributes;
102 }
103 GOFF::EDAttr getEDAttributes() const {
104 assert(isED() && "Not a ED section");
105 return EDAttributes;
106 }
107 GOFF::PRAttr getPRAttributes() const {
108 assert(isPR() && "Not a PR section");
109 return PRAttributes;
110 }
111
112 // Returns the text style for a section. Only defined for ED and PR sections.
113 GOFF::ESDTextStyle getTextStyle() const {
114 assert((isED() || isPR() || isVirtualSection()) && "Expect ED or PR section");
115 if (isED())
116 return EDAttributes.TextStyle;
117 if (isPR())
118 return getParent()->getEDAttributes().TextStyle;
119 // Virtual sections have no data, so byte orientation is fine.
120 return GOFF::ESD_TS_ByteOriented;
121 }
122
123 bool requiresNonZeroLength() const { return RequiresNonZeroLength; }
124
125 void setName(StringRef SectionName) { Name = SectionName; }
126
127 static bool classof(const MCSection *S) { return S->getVariant() == SV_GOFF; }
