45/// The base class of a Sandbox IR Pass.
46class Pass {
47protected:
48 /// The pass name. This is also used as a command-line flag and should not
49 /// contain whitespaces.
50 const std::string Name;
51
52public:
53 /// \p Name can't contain any spaces or start with '-'.
54 Pass(StringRef Name) : Name(Name) {
55 assert(!Name.contains(' ') &&
56 "A pass name should not contain whitespaces!");
57 assert(!Name.starts_with('-') && "A pass name should not start with '-'!");
58 }
59 virtual ~Pass() {}
60 /// \Returns the name of the pass.
61 StringRef getName() const { return Name; }
62#ifndef NDEBUG
63 friend raw_ostream &operator<<(raw_ostream &OS, const Pass &Pass) {
64 Pass.print(OS);
65 return OS;
66 }
67 virtual void print(raw_ostream &OS) const { OS << Name; }
68 LLVM_DUMP_METHOD virtual void dump() const;
69#endif
70 /// Similar to print() but adds a newline. Used for testing.
71 virtual void printPipeline(raw_ostream &OS) const { OS << Name << "\n"; }
72};
