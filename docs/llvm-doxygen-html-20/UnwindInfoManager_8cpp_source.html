67bool UnwindInfoManager::TryEnable() {
68#ifdef __APPLE__
69 static std::mutex M;
70 std::lock_guard<std::mutex> Lock(M);
71
72 if (Instance)
73 return true;
74
75 auto AddFn = (int (*)(void *))dlsym(RTLD_DEFAULT, AddFnName);
76 if (!AddFn)
77 return false;
78
79 auto RemoveFn = (int (*)(void *))dlsym(RTLD_DEFAULT, RemoveFnName);
80 if (!RemoveFn)
81 return false;
82
83 Instance.reset(new UnwindInfoManager());
84
85 if (auto Err = AddFn((void *)&findSections)) {
86 LLVM_DEBUG({
87 dbgs() << "Failed call to " << AddFnName << ": error = " << Err << "\n";
88 });
89 Instance = nullptr;
90 return false;
91 }
92
93 RemoveFindDynamicUnwindSections = RemoveFn;
94 return true;
95
96#else
97 return false;
98#endif // __APPLE__
99}

120int UnwindInfoManager::findSectionsImpl(uintptr_t Addr, UnwindSections *Info) {
121 std::lock_guard<std::mutex> Lock(M);
122 auto I = UWSecs.upper_bound(Addr);
123 if (I == UWSecs.begin())
124 return 0;
125 --I;
126 *Info = I->second;
127 return 1;
128}

130int UnwindInfoManager::findSections(uintptr_t Addr, UnwindSections *Info) {
131 return Instance->findSectionsImpl(Addr, Info);
132}

134Error UnwindInfoManager::registerSectionsImpl(
135 ArrayRef<ExecutorAddrRange> CodeRanges, ExecutorAddr DSOBase,
136 ExecutorAddrRange DWARFEHFrame, ExecutorAddrRange CompactUnwind) {
137 std::lock_guard<std::mutex> Lock(M);
138 for (auto &R : CodeRanges)
139 UWSecs[R.Start.getValue()] =
140 UnwindSections{static_cast<uintptr_t>(DSOBase.getValue()),
141 static_cast<uintptr_t>(DWARFEHFrame.Start.getValue()),
142 static_cast<size_t>(DWARFEHFrame.size()),
143 static_cast<uintptr_t>(CompactUnwind.Start.getValue()),
144 static_cast<size_t>(CompactUnwind.size())};
145 return Error::success();
146}
147
148Error UnwindInfoManager::deregisterSectionsImpl(
149 ArrayRef<ExecutorAddrRange> CodeRanges) {
150 std::lock_guard<std::mutex> Lock(M);
151 for (auto &R : CodeRanges) {
152 auto I = UWSecs.find(R.Start.getValue());
153 if (I == UWSecs.end())
154 return make_error<StringError>(
155 "No unwind-info sections registered for range " +
156 formatv("{0:x} - {1:x}", R.Start, R.End),
157 inconvertibleErrorCode());
158 UWSecs.erase(I);
159 }
160 return Error::success();
161}
