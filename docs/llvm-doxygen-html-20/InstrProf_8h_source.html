582 // Map the MD5 of the symbol name to the name.
583 Error addSymbolName(StringRef SymbolName) {
584 if (SymbolName.empty())
585 return make_error<InstrProfError>(instrprof_error::malformed,
586 "symbol name is empty");
587
588 // Insert into NameTab so that MD5NameMap (a vector that will be sorted)
589 // won't have duplicated entries in the first place.
590 auto Ins = NameTab.insert(SymbolName);
591 if (Ins.second) {
592 MD5NameMap.push_back(std::make_pair(
593 IndexedInstrProf::ComputeHash(SymbolName), Ins.first->getKey()));
594 Sorted = false;
595 }
596 return Error::success();
597 }
