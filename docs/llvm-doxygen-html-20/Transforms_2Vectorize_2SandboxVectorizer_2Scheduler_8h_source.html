9// This is the bottom-up list scheduler used by the vectorizer. It is used for
10// checking the legality of vectorization and for scheduling instructions in
11// such a way that makes vectorization possible, if legal.
12//
13// The legality check is performed by `trySchedule(Instrs)`, which will try to
14// schedule the IR until all instructions in `Instrs` can be scheduled together
15// back-to-back. If this fails then it is illegal to vectorize `Instrs`.
16//
17// Internally the scheduler uses the vectorizer-specific DependencyGraph class.
18//
19//===----------------------------------------------------------------------===//

30class PriorityCmp {
31public:
32 bool operator()(const DGNode *N1, const DGNode *N2) {
33 // TODO: This should be a hierarchical comparator.
34 return N1->getInstruction()->comesBefore(N2->getInstruction());
35 }

38/// The list holding nodes that are ready to schedule. Used by the scheduler.
39class ReadyListContainer {
40 PriorityCmp Cmp;
41 /// Control/Other dependencies are not modeled by the DAG to save memory.
42 /// These have to be modeled in the ready list for correctness.
43 /// This means that the list will hold back nodes that need to meet such
44 /// unmodeled dependencies.
45 std::priority_queue<DGNode *, std::vector<DGNode *>, PriorityCmp> List;
46
47public:
48 ReadyListContainer() : List(Cmp) {}
49 void insert(DGNode *N) { List.push(N); }
50 DGNode *pop() {
51 auto *Back = List.top();
52 List.pop();
53 return Back;
54 }
55 bool empty() const { return List.empty(); }
56 void clear() { List = {}; }
57#ifndef NDEBUG
58 void dump(raw_ostream &OS) const;
59 LLVM_DUMP_METHOD void dump() const;
60#endif // NDEBUG
61};

63/// The nodes that need to be scheduled back-to-back in a single scheduling
64/// cycle form a SchedBundle.
65class SchedBundle {
66public:
67 using ContainerTy = SmallVector<DGNode *, 4>;
68
69private:
70 ContainerTy Nodes;
71
72 /// Called by the DGNode destructor to avoid accessing freed memory.
73 void eraseFromBundle(DGNode *N) { Nodes.erase(find(Nodes, N)); }
74 friend DGNode::~DGNode(); // For eraseFromBundle().
75
76public:
77 SchedBundle() = default;
78 SchedBundle(ContainerTy &&Nodes) : Nodes(std::move(Nodes)) {
79 for (auto *N : this->Nodes)
80 N->setSchedBundle(*this);
81 }
82 /// Copy CTOR (unimplemented).
83 SchedBundle(const SchedBundle &Other) = delete;
84 /// Copy Assignment (unimplemented).
85 SchedBundle &operator=(const SchedBundle &Other) = delete;
86 ~SchedBundle() {
87 for (auto *N : this->Nodes)
88 N->clearSchedBundle();
89 }
90 bool empty() const { return Nodes.empty(); }
91 DGNode *back() const { return Nodes.back(); }
92 using iterator = ContainerTy::iterator;
93 using const_iterator = ContainerTy::const_iterator;
94 iterator begin() { return Nodes.begin(); }
95 iterator end() { return Nodes.end(); }
96 const_iterator begin() const { return Nodes.begin(); }
97 const_iterator end() const { return Nodes.end(); }
98 /// \Returns the bundle node that comes before the others in program order.
99 DGNode *getTop() const;
100 /// \Returns the bundle node that comes after the others in program order.
101 DGNode *getBot() const;
102 /// Move all bundle instructions to \p Where back-to-back.
103 void cluster(BasicBlock::iterator Where);
104#ifndef NDEBUG
105 void dump(raw_ostream &OS) const;
106 LLVM_DUMP_METHOD void dump() const;
107#endif
108};

110/// The list scheduler.
111class Scheduler {
112 /// This is a list-scheduler and this is the list containing the instructions
113 /// that are ready, meaning that all their dependency successors have already
114 /// been scheduled.
115 ReadyListContainer ReadyList;
116 /// The dependency graph is used by the scheduler to determine the legal
117 /// ordering of instructions.
118 DependencyGraph DAG;
119 /// This is the top of the schedule, i.e. the location where the scheduler
120 /// is about to place the scheduled instructions. It gets updated as we
121 /// schedule.
122 std::optional<BasicBlock::iterator> ScheduleTopItOpt;
123 // TODO: This is wasting memory in exchange for fast removal using a raw ptr.
124 DenseMap<SchedBundle *, std::unique_ptr<SchedBundle>> Bndls;
125 /// The BB that we are currently scheduling.
126 BasicBlock *ScheduledBB = nullptr;
127
128 /// \Returns a scheduling bundle containing \p Instrs.
129 SchedBundle *createBundle(ArrayRef<Instruction *> Instrs);
130 void eraseBundle(SchedBundle *SB);
131 /// Schedule nodes until we can schedule \p Instrs back-to-back.
132 bool tryScheduleUntil(ArrayRef<Instruction *> Instrs);
133 /// Schedules all nodes in \p Bndl, marks them as scheduled, updates the
134 /// UnscheduledSuccs counter of all dependency predecessors, and adds any of
135 /// them that become ready to the ready list.
136 void scheduleAndUpdateReadyList(SchedBundle &Bndl);
137 /// The scheduling state of the instructions in the bundle.
138 enum class BndlSchedState {
139 NoneScheduled, ///> No instruction in the bundle was previously scheduled.
140 PartiallyOrDifferentlyScheduled, ///> Only some of the instrs in the bundle
141 /// were previously scheduled, or all of
142 /// them were but not in the same
143 /// SchedBundle.
144 FullyScheduled, ///> All instrs in the bundle were previously scheduled and
145 /// were in the same SchedBundle.
146 };
147 /// \Returns whether none/some/all of \p Instrs have been scheduled.
148 BndlSchedState getBndlSchedState(ArrayRef<Instruction *> Instrs) const;
149 /// Destroy the top-most part of the schedule that includes \p Instrs.
150 void trimSchedule(ArrayRef<Instruction *> Instrs);
151 /// Disable copies.
152 Scheduler(const Scheduler &) = delete;
153 Scheduler &operator=(const Scheduler &) = delete;
154
155public:
156 Scheduler(AAResults &AA, Context &Ctx) : DAG(AA, Ctx) {}
157 ~Scheduler() {}
158 /// Tries to build a schedule that includes all of \p Instrs scheduled at the
159 /// same scheduling cycle. This essentially checks that there are no
160 /// dependencies among \p Instrs. This function may involve scheduling
161 /// intermediate instructions or canceling and re-scheduling if needed.
162 /// \Returns true on success, false otherwise.
163 bool trySchedule(ArrayRef<Instruction *> Instrs);
164 /// Clear the scheduler's state, including the DAG.
165 void clear() {
166 Bndls.clear();
167 // TODO: clear view once it lands.
168 DAG.clear();
169 ReadyList.clear();
170 ScheduleTopItOpt = std::nullopt;
171 ScheduledBB = nullptr;
172 assert(Bndls.empty() && DAG.empty() && ReadyList.empty() &&
173 !ScheduleTopItOpt && ScheduledBB == nullptr &&
174 "Expected empty state!");
175 }
176
177#ifndef NDEBUG
178 void dump(raw_ostream &OS) const;
179 LLVM_DUMP_METHOD void dump() const;
180#endif
181};
