48/// Class to build a trie of call stack contexts for a particular profiled
49/// allocation call, along with their associated allocation types.
50/// The allocation will be at the root of the trie, which is then used to
51/// compute the minimum lists of context ids needed to associate a call context
52/// with a single allocation type.
53class CallStackTrie {
54private:
55 struct CallStackTrieNode {
56 // Allocation types for call context sharing the context prefix at this
57 // node.
58 uint8_t AllocTypes;
59 // If the user has requested reporting of hinted sizes, keep track of the
60 // associated full stack id and profiled sizes. Can have more than one
61 // after trimming (e.g. when building from metadata). This is only placed on
62 // the last (root-most) trie node for each allocation context.
63 std::vector<ContextTotalSize> ContextSizeInfo;
64 // Map of caller stack id to the corresponding child Trie node.
65 std::map<uint64_t, CallStackTrieNode *> Callers;
66 CallStackTrieNode(AllocationType Type)
67 : AllocTypes(static_cast<uint8_t>(Type)) {}
68 void addAllocType(AllocationType AllocType) {
69 AllocTypes |= static_cast<uint8_t>(AllocType);
70 }
71 void removeAllocType(AllocationType AllocType) {
72 AllocTypes &= ~static_cast<uint8_t>(AllocType);
73 }
74 bool hasAllocType(AllocationType AllocType) const {
75 return AllocTypes & static_cast<uint8_t>(AllocType);
76 }
77 };
78
79 // The node for the allocation at the root.
80 CallStackTrieNode *Alloc = nullptr;
81 // The allocation's leaf stack id.
82 uint64_t AllocStackId = 0;
83
84 void deleteTrieNode(CallStackTrieNode *Node) {
85 if (!Node)
86 return;
87 for (auto C : Node->Callers)
88 deleteTrieNode(C.second);
89 delete Node;
90 }

92 // Recursively build up a complete list of context size information from the
93 // trie nodes reached form the given Node, for hint size reporting.
94 void collectContextSizeInfo(CallStackTrieNode *Node,
95 std::vector<ContextTotalSize> &ContextSizeInfo);
96
97 // Recursively convert hot allocation types to notcold, since we don't
98 // actually do any cloning for hot contexts, to facilitate more aggressive
99 // pruning of contexts.
100 void convertHotToNotCold(CallStackTrieNode *Node);
101
102 // Recursive helper to trim contexts and create metadata nodes.
103 bool buildMIBNodes(CallStackTrieNode *Node, LLVMContext &Ctx,
104 std::vector<uint64_t> &MIBCallStack,
105 std::vector<Metadata *> &MIBNodes,
106 bool CalleeHasAmbiguousCallerContext);
107
108public:
109 CallStackTrie() = default;
110 ~CallStackTrie() { deleteTrieNode(Alloc); }
111
112 bool empty() const { return Alloc == nullptr; }
113
114 /// Add a call stack context with the given allocation type to the Trie.
115 /// The context is represented by the list of stack ids (computed during
116 /// matching via a debug location hash), expected to be in order from the
117 /// allocation call down to the bottom of the call stack (i.e. callee to
118 /// caller order).
119 void addCallStack(AllocationType AllocType, ArrayRef<uint64_t> StackIds,
120 std::vector<ContextTotalSize> ContextSizeInfo = {});
121
122 /// Add the call stack context along with its allocation type from the MIB
123 /// metadata to the Trie.
124 void addCallStack(MDNode *MIB);
125
126 /// Build and attach the minimal necessary MIB metadata. If the alloc has a
127 /// single allocation type, add a function attribute instead. The reason for
128 /// adding an attribute in this case is that it matches how the behavior for
129 /// allocation calls will be communicated to lib call simplification after
130 /// cloning or another optimization to distinguish the allocation types,
131 /// which is lower overhead and more direct than maintaining this metadata.
132 /// Returns true if memprof metadata attached, false if not (attribute added).
133 bool buildAndAttachMIBMetadata(CallBase *CI);
134
135 /// Add an attribute for the given allocation type to the call instruction.
136 /// If hinted by reporting is enabled, a message is emitted with the given
137 /// descriptor used to identify the category of single allocation type.
138 void addSingleAllocTypeAttribute(CallBase *CI, AllocationType AT,
139 StringRef Descriptor);
140};

142/// Helper class to iterate through stack ids in both metadata (memprof MIB and
143/// callsite) and the corresponding ThinLTO summary data structures
144/// (CallsiteInfo and MIBInfo). This simplifies implementation of client code
145/// which doesn't need to worry about whether we are operating with IR (Regular
146/// LTO), or summary (ThinLTO).
147template <class NodeT, class IteratorT> class CallStack {
148public:
149 CallStack(const NodeT *N = nullptr) : N(N) {}
150
151 // Implement minimum required methods for range-based for loop.
152 // The default implementation assumes we are operating on ThinLTO data
153 // structures, which have a vector of StackIdIndices. There are specialized
154 // versions provided to iterate through metadata.
155 struct CallStackIterator {
156 const NodeT *N = nullptr;
157 IteratorT Iter;
158 CallStackIterator(const NodeT *N, bool End);
159 uint64_t operator*();
160 bool operator==(const CallStackIterator &rhs) { return Iter == rhs.Iter; }
161 bool operator!=(const CallStackIterator &rhs) { return !(*this == rhs); }
162 void operator++() { ++Iter; }
163 };
164
165 bool empty() const { return N == nullptr; }
166
167 CallStackIterator begin() const;
168 CallStackIterator end() const { return CallStackIterator(N, /*End*/ true); }
169 CallStackIterator beginAfterSharedPrefix(CallStack &Other);
170 uint64_t back() const;
171
172private:
173 const NodeT *N = nullptr;
174};
175
176template <class NodeT, class IteratorT>
177CallStack<NodeT, IteratorT>::CallStackIterator::CallStackIterator(
178 const NodeT *N, bool End)
179 : N(N) {
180 if (!N) {
181 Iter = nullptr;
182 return;
183 }
184 Iter = End ? N->StackIdIndices.end() : N->StackIdIndices.begin();
185}
186
187template <class NodeT, class IteratorT>
188uint64_t CallStack<NodeT, IteratorT>::CallStackIterator::operator*() {
189 assert(Iter != N->StackIdIndices.end());
190 return *Iter;
191}
192
193template <class NodeT, class IteratorT>
194uint64_t CallStack<NodeT, IteratorT>::back() const {
195 assert(N);
196 return N->StackIdIndices.back();
197}
198
199template <class NodeT, class IteratorT>
200typename CallStack<NodeT, IteratorT>::CallStackIterator
201CallStack<NodeT, IteratorT>::begin() const {
202 return CallStackIterator(N, /*End*/ false);
203}

205template <class NodeT, class IteratorT>
206typename CallStack<NodeT, IteratorT>::CallStackIterator
207CallStack<NodeT, IteratorT>::beginAfterSharedPrefix(CallStack &Other) {
208 CallStackIterator Cur = begin();
209 for (CallStackIterator OtherCur = Other.begin();
210 Cur != end() && OtherCur != Other.end(); ++Cur, ++OtherCur)
211 assert(*Cur == *OtherCur);
212 return Cur;
213}
