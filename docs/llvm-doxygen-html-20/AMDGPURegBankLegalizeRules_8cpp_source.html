41bool matchUniformityAndLLT(Register Reg, UniformityLLTOpPredicateID UniID,
42 const MachineUniformityInfo &MUI,
43 const MachineRegisterInfo &MRI) {
44 switch (UniID) {
45 case S1:
46 return MRI.getType(Reg) == LLT::scalar(1);
47 case S16:
48 return MRI.getType(Reg) == LLT::scalar(16);
49 case S32:
50 return MRI.getType(Reg) == LLT::scalar(32);
51 case S64:
52 return MRI.getType(Reg) == LLT::scalar(64);
53 case P1:
54 return MRI.getType(Reg) == LLT::pointer(1, 64);
55 case P3:
56 return MRI.getType(Reg) == LLT::pointer(3, 32);
57 case P4:
58 return MRI.getType(Reg) == LLT::pointer(4, 64);
59 case P5:
60 return MRI.getType(Reg) == LLT::pointer(5, 32);
61 case B32:
62 return MRI.getType(Reg).getSizeInBits() == 32;
63 case B64:
64 return MRI.getType(Reg).getSizeInBits() == 64;
65 case B96:
66 return MRI.getType(Reg).getSizeInBits() == 96;
67 case B128:
68 return MRI.getType(Reg).getSizeInBits() == 128;
69 case B256:
70 return MRI.getType(Reg).getSizeInBits() == 256;
71 case B512:
72 return MRI.getType(Reg).getSizeInBits() == 512;
73 case UniS1:
74 return MRI.getType(Reg) == LLT::scalar(1) && MUI.isUniform(Reg);
75 case UniS16:
76 return MRI.getType(Reg) == LLT::scalar(16) && MUI.isUniform(Reg);
77 case UniS32:
78 return MRI.getType(Reg) == LLT::scalar(32) && MUI.isUniform(Reg);
79 case UniS64:
80 return MRI.getType(Reg) == LLT::scalar(64) && MUI.isUniform(Reg);
81 case UniP1:
82 return MRI.getType(Reg) == LLT::pointer(1, 64) && MUI.isUniform(Reg);
83 case UniP3:
84 return MRI.getType(Reg) == LLT::pointer(3, 32) && MUI.isUniform(Reg);
85 case UniP4:
86 return MRI.getType(Reg) == LLT::pointer(4, 64) && MUI.isUniform(Reg);
87 case UniP5:
88 return MRI.getType(Reg) == LLT::pointer(5, 32) && MUI.isUniform(Reg);
89 case UniB32:
90 return MRI.getType(Reg).getSizeInBits() == 32 && MUI.isUniform(Reg);
91 case UniB64:
92 return MRI.getType(Reg).getSizeInBits() == 64 && MUI.isUniform(Reg);
93 case UniB96:
94 return MRI.getType(Reg).getSizeInBits() == 96 && MUI.isUniform(Reg);
95 case UniB128:
96 return MRI.getType(Reg).getSizeInBits() == 128 && MUI.isUniform(Reg);
97 case UniB256:
98 return MRI.getType(Reg).getSizeInBits() == 256 && MUI.isUniform(Reg);
99 case UniB512:
100 return MRI.getType(Reg).getSizeInBits() == 512 && MUI.isUniform(Reg);
101 case DivS1:
102 return MRI.getType(Reg) == LLT::scalar(1) && MUI.isDivergent(Reg);
103 case DivS32:
104 return MRI.getType(Reg) == LLT::scalar(32) && MUI.isDivergent(Reg);
105 case DivS64:
106 return MRI.getType(Reg) == LLT::scalar(64) && MUI.isDivergent(Reg);
107 case DivP1:
108 return MRI.getType(Reg) == LLT::pointer(1, 64) && MUI.isDivergent(Reg);
109 case DivP3:
110 return MRI.getType(Reg) == LLT::pointer(3, 32) && MUI.isDivergent(Reg);
111 case DivP4:
112 return MRI.getType(Reg) == LLT::pointer(4, 64) && MUI.isDivergent(Reg);
113 case DivP5:
114 return MRI.getType(Reg) == LLT::pointer(5, 32) && MUI.isDivergent(Reg);
115 case DivB32:
116 return MRI.getType(Reg).getSizeInBits() == 32 && MUI.isDivergent(Reg);
117 case DivB64:
118 return MRI.getType(Reg).getSizeInBits() == 64 && MUI.isDivergent(Reg);
119 case DivB96:
120 return MRI.getType(Reg).getSizeInBits() == 96 && MUI.isDivergent(Reg);
121 case DivB128:
122 return MRI.getType(Reg).getSizeInBits() == 128 && MUI.isDivergent(Reg);
123 case DivB256:
124 return MRI.getType(Reg).getSizeInBits() == 256 && MUI.isDivergent(Reg);
125 case DivB512:
126 return MRI.getType(Reg).getSizeInBits() == 512 && MUI.isDivergent(Reg);
127 case _:
128 return true;
129 default:
130 llvm_unreachable("missing matchUniformityAndLLT");
131 }

134bool PredicateMapping::match(const MachineInstr &MI,
135 const MachineUniformityInfo &MUI,
136 const MachineRegisterInfo &MRI) const {
137 // Check LLT signature.
138 for (unsigned i = 0; i < OpUniformityAndTypes.size(); ++i) {
139 if (OpUniformityAndTypes[i] == _) {
140 if (MI.getOperand(i).isReg())
141 return false;
142 continue;
143 }
144
145 // Remaining IDs check registers.
146 if (!MI.getOperand(i).isReg())
147 return false;
148
149 if (!matchUniformityAndLLT(MI.getOperand(i).getReg(),
150 OpUniformityAndTypes[i], MUI, MRI))
151 return false;
152 }
153
154 // More complex check.
155 if (TestFunc)
156 return TestFunc(MI);
157
158 return true;
159}

166UniformityLLTOpPredicateID LLTToId(LLT Ty) {
167 if (Ty == LLT::scalar(16))
168 return S16;
169 if (Ty == LLT::scalar(32))
170 return S32;
171 if (Ty == LLT::scalar(64))
172 return S64;
173 if (Ty == LLT::fixed_vector(2, 16))
174 return V2S16;
175 if (Ty == LLT::fixed_vector(2, 32))
176 return V2S32;
177 if (Ty == LLT::fixed_vector(3, 32))
178 return V3S32;
179 if (Ty == LLT::fixed_vector(4, 32))
180 return V4S32;
181 return _;
182}
183
184UniformityLLTOpPredicateID LLTToBId(LLT Ty) {
185 if (Ty == LLT::scalar(32) || Ty == LLT::fixed_vector(2, 16) ||
186 Ty == LLT::pointer(3, 32) || Ty == LLT::pointer(5, 32) ||
187 Ty == LLT::pointer(6, 32))
188 return B32;
189 if (Ty == LLT::scalar(64) || Ty == LLT::fixed_vector(2, 32) ||
190 Ty == LLT::fixed_vector(4, 16) || Ty == LLT::pointer(1, 64) ||
191 Ty == LLT::pointer(4, 64))
192 return B64;
193 if (Ty == LLT::fixed_vector(3, 32))
194 return B96;
195 if (Ty == LLT::fixed_vector(4, 32))
196 return B128;
197 return _;
198}

200const RegBankLLTMapping &
201SetOfRulesForOpcode::findMappingForMI(const MachineInstr &MI,
202 const MachineRegisterInfo &MRI,
203 const MachineUniformityInfo &MUI) const {
204 // Search in "Fast Rules".
205 // Note: if fast rules are enabled, RegBankLLTMapping must be added in each
206 // slot that could "match fast Predicate". If not, InvalidMapping is
207 // returned which results in failure, does not search "Slow Rules".
208 if (FastTypes != NoFastRules) {
209 Register Reg = MI.getOperand(0).getReg();
210 int Slot;
211 if (FastTypes == StandardB)
212 Slot = getFastPredicateSlot(LLTToBId(MRI.getType(Reg)));
213 else
214 Slot = getFastPredicateSlot(LLTToId(MRI.getType(Reg)));
215
216 if (Slot != -1)
217 return MUI.isUniform(Reg) ? Uni[Slot] : Div[Slot];
218 }
219
220 // Slow search for more complex rules.
221 for (const RegBankLegalizeRule &Rule : Rules) {
222 if (Rule.Predicate.match(MI, MUI, MRI))
223 return Rule.OperandMapping;
224 }
225
226 LLVM_DEBUG(dbgs() << "MI: "; MI.dump(););
227 llvm_unreachable("None of the rules defined for MI's opcode matched MI");
228}

248int SetOfRulesForOpcode::getFastPredicateSlot(
249 UniformityLLTOpPredicateID Ty) const {
250 switch (FastTypes) {
251 case Standard: {
252 switch (Ty) {
253 case S32:
254 return 0;
255 case S16:
256 return 1;
257 case S64:
258 return 2;
259 case V2S16:
260 return 3;
261 default:
262 return -1;
263 }
264 }
265 case StandardB: {
266 switch (Ty) {
267 case B32:
268 return 0;
269 case B64:
270 return 1;
271 case B96:
272 return 2;
273 case B128:
274 return 3;
275 default:
276 return -1;
277 }
278 }
279 case Vector: {
280 switch (Ty) {
281 case S32:
282 return 0;
283 case V2S32:
284 return 1;
285 case V3S32:
286 return 2;
287 case V4S32:
288 return 3;
289 default:
290 return -1;
291 }
292 }
293 default:
294 return -1;
295 }

310const SetOfRulesForOpcode &
311RegBankLegalizeRules::getRulesForOpc(MachineInstr &MI) const {
312 unsigned Opc = MI.getOpcode();
313 if (Opc == AMDGPU::G_INTRINSIC || Opc == AMDGPU::G_INTRINSIC_CONVERGENT ||
314 Opc == AMDGPU::G_INTRINSIC_W_SIDE_EFFECTS ||
315 Opc == AMDGPU::G_INTRINSIC_CONVERGENT_W_SIDE_EFFECTS) {
316 unsigned IntrID = cast<GIntrinsic>(MI).getIntrinsicID();
317 if (!IRulesAlias.contains(IntrID)) {
318 LLVM_DEBUG(dbgs() << "MI: "; MI.dump(););
319 llvm_unreachable("No rules defined for intrinsic opcode");
320 }
321 return IRules.at(IRulesAlias.at(IntrID));
322 }
323
324 if (!GRulesAlias.contains(Opc)) {
325 LLVM_DEBUG(dbgs() << "MI: "; MI.dump(););
326 llvm_unreachable("No rules defined for generic opcode");
327 }
328 return GRules.at(GRulesAlias.at(Opc));
329}

331// Syntactic sugar wrapper for predicate lambda that enables '&&', '||' and '!'.
332class Predicate {
333private:
334 struct Elt {
335 // Save formula composed of Pred, '&&', '||' and '!' as a jump table.
336 // Sink ! to Pred. For example !((A && !B) || C) -> (!A || B) && !C
337 // Sequences of && and || will be represented by jumps, for example:
338 // (A && B && ... X) or (A && B && ... X) || Y
339 // A == true jump to B
340 // A == false jump to end or Y, result is A(false) or Y
341 // (A || B || ... X) or (A || B || ... X) && Y
342 // A == true jump to end or Y, result is A(true) or Y
343 // A == false jump to B
344 // Notice that when negating expression, we simply flip Neg on each Pred
345 // and swap TJumpOffset and FJumpOffset (&& becomes ||, || becomes &&).
346 std::function<bool(const MachineInstr &)> Pred;
347 bool Neg; // Neg of Pred is calculated before jump
348 unsigned TJumpOffset;
349 unsigned FJumpOffset;
350 };
351
352 SmallVector<Elt, 8> Expression;
353
354 Predicate(SmallVectorImpl<Elt> &&Expr) { Expression.swap(Expr); };
355
356public:
357 Predicate(std::function<bool(const MachineInstr &)> Pred) {
358 Expression.push_back({Pred, false, 1, 1});
359 };
360
361 bool operator()(const MachineInstr &MI) const {
362 unsigned Idx = 0;
363 unsigned ResultIdx = Expression.size();
364 bool Result;
365 do {
366 Result = Expression[Idx].Pred(MI);
367 Result = Expression[Idx].Neg ? !Result : Result;
368 if (Result) {
369 Idx += Expression[Idx].TJumpOffset;
370 } else {
371 Idx += Expression[Idx].FJumpOffset;
372 }
373 } while ((Idx != ResultIdx));
374
375 return Result;
376 };
377
378 Predicate operator!() const {
379 SmallVector<Elt, 8> NegExpression;
380 for (const Elt &ExprElt : Expression) {
381 NegExpression.push_back({ExprElt.Pred, !ExprElt.Neg, ExprElt.FJumpOffset,
382 ExprElt.TJumpOffset});
383 }
384 return Predicate(std::move(NegExpression));
385 };
386
387 Predicate operator&&(const Predicate &RHS) const {
388 SmallVector<Elt, 8> AndExpression = Expression;
389
390 unsigned RHSSize = RHS.Expression.size();
391 unsigned ResultIdx = Expression.size();
392 for (unsigned i = 0; i < ResultIdx; ++i) {
393 // LHS results in false, whole expression results in false.
394 if (i + AndExpression[i].FJumpOffset == ResultIdx)
395 AndExpression[i].FJumpOffset += RHSSize;
396 }
397
398 AndExpression.append(RHS.Expression);
399
400 return Predicate(std::move(AndExpression));
401 }
402
403 Predicate operator||(const Predicate &RHS) const {
404 SmallVector<Elt, 8> OrExpression = Expression;
405
406 unsigned RHSSize = RHS.Expression.size();
407 unsigned ResultIdx = Expression.size();
408 for (unsigned i = 0; i < ResultIdx; ++i) {
409 // LHS results in true, whole expression results in true.
410 if (i + OrExpression[i].TJumpOffset == ResultIdx)
411 OrExpression[i].TJumpOffset += RHSSize;
412 }
413
414 OrExpression.append(RHS.Expression);
415
416 return Predicate(std::move(OrExpression));
417 }
418};
