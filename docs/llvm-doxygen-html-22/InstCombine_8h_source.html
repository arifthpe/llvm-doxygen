81//
82// InstructionCombining - Combine instructions to form fewer, simple
83// instructions. This pass does not modify the CFG, and has a tendency to make
84// instructions dead, so a subsequent DCE pass is useful.
85//
86// This pass combines things like:
87// %Y = add int 1, %X
88// %Z = add int 1, %Y
89// into:
90// %Z = add int 2, %X
91//
92LLVM_ABI FunctionPass *createInstructionCombiningPass();
93}
