202 unsigned getArgumentNumber() const {
203 assert(Kind == Argument || Kind == ExtendArgument ||
204 Kind == TruncArgument || Kind == SameVecWidthArgument ||
205 Kind == VecElementArgument || Kind == Subdivide2Argument ||
206 Kind == Subdivide4Argument || Kind == VecOfBitcastsToInt);
207 return Argument_Info >> 3;
208 }
209 ArgKind getArgumentKind() const {
210 assert(Kind == Argument || Kind == ExtendArgument ||
211 Kind == TruncArgument || Kind == SameVecWidthArgument ||
212 Kind == VecElementArgument || Kind == Subdivide2Argument ||
213 Kind == Subdivide4Argument || Kind == VecOfBitcastsToInt);
214 return (ArgKind)(Argument_Info & 7);
215 }
216
217 // VecOfAnyPtrsToElt uses both an overloaded argument (for address space)
218 // and a reference argument (for matching vector width and element types)
219 unsigned getOverloadArgNumber() const {
220 assert(Kind == VecOfAnyPtrsToElt);
221 return Argument_Info >> 16;
222 }
223 // OneNthEltsVecArguments uses both a divisor N and a reference argument for
224 // the full-width vector to match
225 unsigned getVectorDivisor() const {
226 assert(Kind == OneNthEltsVecArgument);
227 return Argument_Info >> 16;
228 }
229 unsigned getRefArgNumber() const {
230 assert(Kind == VecOfAnyPtrsToElt || Kind == OneNthEltsVecArgument);
231 return Argument_Info & 0xFFFF;
232 }
