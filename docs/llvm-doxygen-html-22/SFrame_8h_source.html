107template <endianness E> struct FDEInfo {
108 detail::packed<uint8_t, E> Info;
109
110 uint8_t getPAuthKey() const { return (Info >> 5) & 1; }
111 FDEType getFDEType() const { return static_cast<FDEType>((Info >> 4) & 1); }
112 FREType getFREType() const { return static_cast<FREType>(Info & 0xf); }
113 void setPAuthKey(uint8_t P) { setFuncInfo(P, getFDEType(), getFREType()); }
114 void setFDEType(FDEType D) { setFuncInfo(getPAuthKey(), D, getFREType()); }
115 void setFREType(FREType R) { setFuncInfo(getPAuthKey(), getFDEType(), R); }
116 void setFuncInfo(uint8_t PAuthKey, FDEType FDE, FREType FRE) {
117 Info = ((PAuthKey & 1) << 5) | ((static_cast<uint8_t>(FDE) & 1) << 4) |
118 (static_cast<uint8_t>(FRE) & 0xf);
119 }
120 uint8_t getFuncInfo() const { return Info; }
121};

133template <endianness E> struct FREInfo {
134 detail::packed<uint8_t, E> Info;
135
136 bool isReturnAddressSigned() const { return Info >> 7; }
137 FREOffset getOffsetSize() const {
138 return static_cast<FREOffset>((Info >> 5) & 3);
139 }
140 uint8_t getOffsetCount() const { return (Info >> 1) & 0xf; }
141 BaseReg getBaseRegister() const { return static_cast<BaseReg>(Info & 1); }
142 void setReturnAddressSigned(bool RA) {
143 setFREInfo(RA, getOffsetSize(), getOffsetCount(), getBaseRegister());
144 }
145 void setOffsetSize(FREOffset Sz) {
146 setFREInfo(isReturnAddressSigned(), Sz, getOffsetCount(),
147 getBaseRegister());
148 }
149 void setOffsetCount(uint8_t N) {
150 setFREInfo(isReturnAddressSigned(), getOffsetSize(), N, getBaseRegister());
151 }
152 void setBaseRegister(BaseReg Reg) {
153 setFREInfo(isReturnAddressSigned(), getOffsetSize(), getOffsetCount(), Reg);
154 }
155 void setFREInfo(bool RA, FREOffset Sz, uint8_t N, BaseReg Reg) {
156 Info = ((RA & 1) << 7) | ((static_cast<uint8_t>(Sz) & 3) << 5) |
157 ((N & 0xf) << 1) | (static_cast<uint8_t>(Reg) & 1);
158 }
159 uint8_t getFREInfo() const { return Info; }
160};
