294/// This callable defines the behavior of a ThinLTO backend after the thin-link
295/// phase. It accepts a configuration \p C, a combined module summary index
296/// \p CombinedIndex, a map of module identifiers to global variable summaries
297/// \p ModuleToDefinedGVSummaries, a function to add output streams \p
298/// AddStream, and a file cache \p Cache. It returns a unique pointer to a
299/// ThinBackendProc, which can be used to launch backends in parallel.
300using ThinBackendFunction = std::function<std::unique_ptr<ThinBackendProc>(
301 const Config &C, ModuleSummaryIndex &CombinedIndex,
302 const DenseMap<StringRef, GVSummaryMapTy> &ModuleToDefinedGVSummaries,
303 AddStreamFn AddStream, FileCache Cache)>;

305/// This type defines the behavior following the thin-link phase during ThinLTO.
306/// It encapsulates a backend function and a strategy for thread pool
307/// parallelism. Clients should use one of the provided create*ThinBackend()
308/// functions to instantiate a ThinBackend. Parallelism defines the thread pool
309/// strategy to be used for processing.
310struct ThinBackend {
311 ThinBackend(ThinBackendFunction Func, ThreadPoolStrategy Parallelism)
312 : Func(std::move(Func)), Parallelism(std::move(Parallelism)) {}
313 ThinBackend() = default;
314
315 std::unique_ptr<ThinBackendProc> operator()(
316 const Config &Conf, ModuleSummaryIndex &CombinedIndex,
317 const DenseMap<StringRef, GVSummaryMapTy> &ModuleToDefinedGVSummaries,
318 AddStreamFn AddStream, FileCache Cache) {
319 assert(isValid() && "Invalid backend function");
320 return Func(Conf, CombinedIndex, ModuleToDefinedGVSummaries,
321 std::move(AddStream), std::move(Cache));
322 }
323 ThreadPoolStrategy getParallelism() const { return Parallelism; }
324 bool isValid() const { return static_cast<bool>(Func); }
325
326private:
327 ThinBackendFunction Func = nullptr;
328 ThreadPoolStrategy Parallelism;
329};

391/// This class implements a resolution-based interface to LLVM's LTO
392/// functionality. It supports regular LTO, parallel LTO code generation and
393/// ThinLTO. You can use it from a linker in the following way:
394/// - Set hooks and code generation options (see lto::Config struct defined in
395/// Config.h), and use the lto::Config object to create an lto::LTO object.
396/// - Create lto::InputFile objects using lto::InputFile::create(), then use
397/// the symbols() function to enumerate its symbols and compute a resolution
398/// for each symbol (see SymbolResolution below).
399/// - After the linker has visited each input file (and each regular object
400/// file) and computed a resolution for each symbol, take each lto::InputFile
401/// and pass it and an array of symbol resolutions to the add() function.
402/// - Call the getMaxTasks() function to get an upper bound on the number of
403/// native object files that LTO may add to the link.
404/// - Call the run() function. This function will use the supplied AddStream
405/// and Cache functions to add up to getMaxTasks() native object files to
406/// the link.
407class LTO {
408 friend InputFile;
409
410public:

412 enum LTOKind {
413 /// Any LTO mode without Unified LTO. The default mode.
414 LTOK_Default,
415
416 /// Regular LTO, with Unified LTO enabled.
417 LTOK_UnifiedRegular,
418
419 /// ThinLTO, with Unified LTO enabled.
420 LTOK_UnifiedThin,
421 };

524 // The global resolution for a particular (mangled) symbol name. This is in
525 // particular necessary to track whether each symbol can be internalized.
526 // Because any input file may introduce a new cross-partition reference, we
527 // cannot make any final internalization decisions until all input files have
528 // been added and the client has called run(). During run() we apply
529 // internalization decisions either directly to the module (for regular LTO)
530 // or to the combined index (for ThinLTO).
531 struct GlobalResolution {
532 /// The unmangled name of the global.
533 std::string IRName;
534
535 /// Keep track if the symbol is visible outside of a module with a summary
536 /// (i.e. in either a regular object or a regular LTO module without a
537 /// summary).
538 bool VisibleOutsideSummary = false;
539
540 /// The symbol was exported dynamically, and therefore could be referenced
541 /// by a shared library not visible to the linker.
542 bool ExportDynamic = false;
543
544 bool UnnamedAddr = true;
545
546 /// True if module contains the prevailing definition.
547 bool Prevailing = false;
548
549 /// Returns true if module contains the prevailing definition and symbol is
550 /// an IR symbol. For example when module-level inline asm block is used,
551 /// symbol can be prevailing in module but have no IR name.
552 bool isPrevailingIRSymbol() const { return Prevailing && !IRName.empty(); }
553
554 /// This field keeps track of the partition number of this global. The
555 /// regular LTO object is partition 0, while each ThinLTO object has its own
556 /// partition number from 1 onwards.
557 ///
558 /// Any global that is defined or used by more than one partition, or that
559 /// is referenced externally, may not be internalized.
560 ///
561 /// Partitions generally have a one-to-one correspondence with tasks, except
562 /// that we use partition 0 for all parallel LTO code generation partitions.
563 /// Any partitioning of the combined LTO object is done internally by the
564 /// LTO backend.
565 unsigned Partition = Unknown;
566
567 /// Special partition numbers.
568 enum : unsigned {
569 /// A partition number has not yet been assigned to this global.
570 Unknown = -1u,
571
572 /// This global is either used by more than one partition or has an
573 /// external reference, and therefore cannot be internalized.
574 External = -2u,
575
576 /// The RegularLTO partition
577 RegularLTO = 0,
578 };
579 };
