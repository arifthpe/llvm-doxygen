164/// Performs placeholder substitution in dynamic library paths.
165///
166/// Configures known placeholders (like @loader_path) and replaces them
167/// in input paths with their resolved values.
168class DylibSubstitutor {
169public:
170 void configure(StringRef loaderPath);
171
172 std::string substitute(StringRef input) const {
173 for (const auto &[ph, value] : Placeholders) {
174 if (input.starts_with_insensitive(ph))
175 return (Twine(value) + input.drop_front(ph.size())).str();
176 }
177 return input.str();
178 }
179
180private:
181 SmallVector<std::pair<std::string, std::string>> Placeholders;
182};

184/// Loads an object file and provides access to it.
185///
186/// Owns the underlying `ObjectFile` and ensures it is valid.
187/// Any errors encountered during construction are stored and
188/// returned when attempting to access the file.
189class ObjectFileLoader {
190public:
191 /// Construct an object file loader from the given path.
192 explicit ObjectFileLoader(StringRef Path) {
193 auto ObjOrErr = loadObjectFileWithOwnership(Path);
194 if (ObjOrErr)
195 Obj = std::move(*ObjOrErr);
196 else {
197 consumeError(std::move(Err));
198 Err = ObjOrErr.takeError();
199 }
200 }
201
202 ObjectFileLoader(const ObjectFileLoader &) = delete;
203 ObjectFileLoader &operator=(const ObjectFileLoader &) = delete;
204
205 ObjectFileLoader(ObjectFileLoader &&) = default;
206 ObjectFileLoader &operator=(ObjectFileLoader &&) = default;
207
208 /// Get the loaded object file, or return an error if loading failed.
209 Expected<object::ObjectFile &> getObjectFile() {
210 if (Err) {
211 // allow the error to be taken only once
212 if (ErrorTaken)
213 return createStringError(inconvertibleErrorCode(),
214 "error already taken");
215
216 ErrorTaken = true;
217 return std::move(Err);
218 }
219 return *Obj.getBinary();
220 }
221
222 static bool isArchitectureCompatible(const object::ObjectFile &Obj);
223
224private:
225 object::OwningBinary<object::ObjectFile> Obj;
226 Error Err = Error::success();
227 bool ErrorTaken = false;
228
229 static Expected<object::OwningBinary<object::ObjectFile>>
230 loadObjectFileWithOwnership(StringRef FilePath);
231};

258/// Validates and normalizes dynamic library paths.
259///
260/// Uses a `PathResolver` to resolve paths to their canonical form and
261/// checks whether they point to valid shared libraries.
262class DylibPathValidator {
263public:
264 DylibPathValidator(PathResolver &PR, LibraryPathCache &LC,
265 ObjFileCache *ObjCache = nullptr)
266 : LibPathResolver(PR), LibPathCache(LC), ObjCache(ObjCache) {}
267
268 bool isSharedLibrary(StringRef Path) const;
269 bool isSharedLibraryCached(StringRef Path) const {
270 if (LibPathCache.hasSeen(Path))
271 return true;
272 return isSharedLibrary(Path);
273 }
274
275 std::optional<std::string> normalize(StringRef Path) const {
276 std::error_code ec;
277 auto real = LibPathResolver.resolve(Path, ec);
278 if (!real || ec)
279 return std::nullopt;
280
281 return real;
282 }
283
284 /// Validate the given path as a shared library.
285 std::optional<std::string> validate(StringRef Path) const {
286 if (LibPathCache.hasSeen(Path))
287 return Path.str();
288 auto realOpt = normalize(Path);
289 if (!realOpt)
290 return std::nullopt;
291
292 if (!isSharedLibraryCached(*realOpt))
293 return std::nullopt;
294
295 return realOpt;
296 }
297
298private:
299 PathResolver &LibPathResolver;
300 LibraryPathCache &LibPathCache;
301 mutable ObjFileCache *ObjCache;
302};
