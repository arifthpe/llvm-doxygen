32template <typename FailT, typename Fun, typename... Args>
33inline decltype(auto) RetryAfterSignal(const FailT &Fail, const Fun &F,
34 const Args &... As) {
35 decltype(F(As...)) Res;
36 do {
37 errno = 0;
38 Res = F(As...);
39 } while (Res == Fail && errno == EINTR);
40 return Res;
41}
