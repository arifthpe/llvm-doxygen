44/// isUsedOutsideOfDefiningBlock - Return true if this instruction is used by
45/// PHI nodes or outside of the basic block that defines it, or used by a
46/// switch or atomic instruction, which may expand to multiple basic blocks.
47static bool isUsedOutsideOfDefiningBlock(const Instruction *I) {
48 if (I->use_empty()) return false;
49 if (isa<PHINode>(I)) return true;
50 const BasicBlock *BB = I->getParent();
51 for (const User *U : I->users())
52 if (cast<Instruction>(U)->getParent() != BB || isa<PHINode>(U))
53 return true;
54
55 return false;
56}

58static ISD::NodeType getPreferredExtendForValue(const Instruction *I) {
59 // For the users of the source value being used for compare instruction, if
60 // the number of signed predicate is greater than unsigned predicate, we
61 // prefer to use SIGN_EXTEND.
62 //
63 // With this optimization, we would be able to reduce some redundant sign or
64 // zero extension instruction, and eventually more machine CSE opportunities
65 // can be exposed.
66 ISD::NodeType ExtendKind = ISD::ANY_EXTEND;
67 unsigned NumOfSigned = 0, NumOfUnsigned = 0;
68 for (const Use &U : I->uses()) {
69 if (const auto *CI = dyn_cast<CmpInst>(U.getUser())) {
70 NumOfSigned += CI->isSigned();
71 NumOfUnsigned += CI->isUnsigned();
72 }
73 if (const auto *CallI = dyn_cast<CallBase>(U.getUser())) {
74 if (!CallI->isArgOperand(&U))
75 continue;
76 unsigned ArgNo = CallI->getArgOperandNo(&U);
77 NumOfUnsigned += CallI->paramHasAttr(ArgNo, Attribute::ZExt);
78 NumOfSigned += CallI->paramHasAttr(ArgNo, Attribute::SExt);
79 }
80 }
81 if (NumOfSigned > NumOfUnsigned)
82 ExtendKind = ISD::SIGN_EXTEND;
83
84 return ExtendKind;
85}

87void FunctionLoweringInfo::set(const Function &fn, MachineFunction &mf,
88 SelectionDAG *DAG) {
89 Fn = &fn;
90 MF = &mf;
91 TLI = MF->getSubtarget().getTargetLowering();
92 RegInfo = &MF->getRegInfo();
93 const TargetFrameLowering *TFI = MF->getSubtarget().getFrameLowering();
94 UA = DAG->getUniformityInfo();
95
96 // Check whether the function can return without sret-demotion.
97 SmallVector<ISD::OutputArg, 4> Outs;
98 CallingConv::ID CC = Fn->getCallingConv();
99
100 GetReturnInfo(CC, Fn->getReturnType(), Fn->getAttributes(), Outs, *TLI,
101 mf.getDataLayout());
102 CanLowerReturn =
103 TLI->CanLowerReturn(CC, *MF, Fn->isVarArg(), Outs, Fn->getContext(), Fn->getReturnType());
104
105 // If this personality uses funclets, we need to do a bit more work.
106 DenseMap<const AllocaInst *, TinyPtrVector<int *>> CatchObjects;
107 EHPersonality Personality = classifyEHPersonality(
108 Fn->hasPersonalityFn() ? Fn->getPersonalityFn() : nullptr);
109 if (isFuncletEHPersonality(Personality)) {
110 // Calculate state numbers if we haven't already.
111 WinEHFuncInfo &EHInfo = *MF->getWinEHFuncInfo();
112 if (Personality == EHPersonality::MSVC_CXX)
113 calculateWinCXXEHStateNumbers(&fn, EHInfo);
114 else if (isAsynchronousEHPersonality(Personality))
115 calculateSEHStateNumbers(&fn, EHInfo);
116 else if (Personality == EHPersonality::CoreCLR)
117 calculateClrEHStateNumbers(&fn, EHInfo);
118
119 // Map all BB references in the WinEH data to MBBs.
120 for (WinEHTryBlockMapEntry &TBME : EHInfo.TryBlockMap) {
121 for (WinEHHandlerType &H : TBME.HandlerArray) {
122 if (const AllocaInst *AI = H.CatchObj.Alloca)
123 CatchObjects[AI].push_back(&H.CatchObj.FrameIndex);
124 else
125 H.CatchObj.FrameIndex = INT_MAX;
126 }
127 }
128 }
129
130 // Initialize the mapping of values to registers. This is only set up for
131 // instruction values that are used outside of the block that defines
132 // them.
133 const Align StackAlign = TFI->getStackAlign();
134 for (const BasicBlock &BB : *Fn) {
135 for (const Instruction &I : BB) {
136 if (const AllocaInst *AI = dyn_cast<AllocaInst>(&I)) {
137 Type *Ty = AI->getAllocatedType();
138 Align Alignment = AI->getAlign();
139
140 // Static allocas can be folded into the initial stack frame
141 // adjustment. For targets that don't realign the stack, don't
142 // do this if there is an extra alignment requirement.
143 if (AI->isStaticAlloca() &&
144 (TFI->isStackRealignable() || (Alignment <= StackAlign))) {
145 const ConstantInt *CUI = cast<ConstantInt>(AI->getArraySize());
146 uint64_t TySize =
147 MF->getDataLayout().getTypeAllocSize(Ty).getKnownMinValue();
148
149 TySize *= CUI->getZExtValue(); // Get total allocated size.
150 if (TySize == 0) TySize = 1; // Don't create zero-sized stack objects.
151 int FrameIndex = INT_MAX;
152 auto Iter = CatchObjects.find(AI);
153 if (Iter != CatchObjects.end() && TLI->needsFixedCatchObjects()) {
154 FrameIndex = MF->getFrameInfo().CreateFixedObject(
155 TySize, 0, /*IsImmutable=*/false, /*isAliased=*/true);
156 MF->getFrameInfo().setObjectAlignment(FrameIndex, Alignment);
157 } else {
158 FrameIndex = MF->getFrameInfo().CreateStackObject(TySize, Alignment,
159 false, AI);
160 }
161
162 // Scalable vectors and structures that contain scalable vectors may
163 // need a special StackID to distinguish them from other (fixed size)
164 // stack objects.
165 if (Ty->isScalableTy())
166 MF->getFrameInfo().setStackID(FrameIndex,
167 TFI->getStackIDForScalableVectors());
168
169 StaticAllocaMap[AI] = FrameIndex;
170 // Update the catch handler information.
171 if (Iter != CatchObjects.end()) {
172 for (int *CatchObjPtr : Iter->second)
173 *CatchObjPtr = FrameIndex;
174 }
175 } else {
176 // FIXME: Overaligned static allocas should be grouped into
177 // a single dynamic allocation instead of using a separate
178 // stack allocation for each one.
179 // Inform the Frame Information that we have variable-sized objects.
180 MF->getFrameInfo().CreateVariableSizedObject(
181 Alignment <= StackAlign ? Align(1) : Alignment, AI);
182 }
183 } else if (auto *Call = dyn_cast<CallBase>(&I)) {
184 // Look for inline asm that clobbers the SP register.
185 if (Call->isInlineAsm()) {
186 Register SP = TLI->getStackPointerRegisterToSaveRestore();
187 const TargetRegisterInfo *TRI = MF->getSubtarget().getRegisterInfo();
188 std::vector<TargetLowering::AsmOperandInfo> Ops =
189 TLI->ParseConstraints(Fn->getDataLayout(), TRI,
190 *Call);
191 for (TargetLowering::AsmOperandInfo &Op : Ops) {
192 if (Op.Type == InlineAsm::isClobber) {
193 // Clobbers don't have SDValue operands, hence SDValue().
194 TLI->ComputeConstraintToUse(Op, SDValue(), DAG);
195 std::pair<unsigned, const TargetRegisterClass *> PhysReg =
196 TLI->getRegForInlineAsmConstraint(TRI, Op.ConstraintCode,
197 Op.ConstraintVT);
198 if (PhysReg.first == SP)
199 MF->getFrameInfo().setHasOpaqueSPAdjustment(true);
200 }
201 }
202 }
203 if (const auto *II = dyn_cast<IntrinsicInst>(&I)) {
204 switch (II->getIntrinsicID()) {
205 case Intrinsic::vastart:
206 // Look for calls to the @llvm.va_start intrinsic. We can omit
207 // some prologue boilerplate for variadic functions that don't
208 // examine their arguments.
209 MF->getFrameInfo().setHasVAStart(true);
210 break;
211 case Intrinsic::fake_use:
212 // Look for llvm.fake.uses, so that we can remove loads into fake
213 // uses later if necessary.
214 MF->setHasFakeUses(true);
215 break;
216 default:
217 break;
218 }
219 }
220
221 // If we have a musttail call in a variadic function, we need to ensure
222 // we forward implicit register parameters.
223 if (const auto *CI = dyn_cast<CallInst>(&I)) {
224 if (CI->isMustTailCall() && Fn->isVarArg())
225 MF->getFrameInfo().setHasMustTailInVarArgFunc(true);
226 }
227
228 // Determine if there is a call to setjmp in the machine function.
229 if (Call->hasFnAttr(Attribute::ReturnsTwice))
230 MF->setExposesReturnsTwice(true);
231 }
232
233 // Mark values used outside their block as exported, by allocating
234 // a virtual register for them.
235 if (isUsedOutsideOfDefiningBlock(&I))
236 if (!isa<AllocaInst>(I) || !StaticAllocaMap.count(cast<AllocaInst>(&I)))
237 InitializeRegForValue(&I);
238
239 // Decide the preferred extend type for a value. This iterates over all
240 // users and therefore isn't cheap, so don't do this at O0.
241 if (DAG->getOptLevel() != CodeGenOptLevel::None)
242 PreferredExtendType[&I] = getPreferredExtendForValue(&I);
243 }
244 }
245
246 // Create an initial MachineBasicBlock for each LLVM BasicBlock in F. This
247 // also creates the initial PHI MachineInstrs, though none of the input
248 // operands are populated.
249 MBBMap.resize(Fn->getMaxBlockNumber());
250 for (const BasicBlock &BB : *Fn) {
251 // Don't create MachineBasicBlocks for imaginary EH pad blocks. These blocks
252 // are really data, and no instructions can live here.
253 if (BB.isEHPad()) {
254 BasicBlock::const_iterator PadInst = BB.getFirstNonPHIIt();
255 // If this is a non-landingpad EH pad, mark this function as using
256 // funclets.
257 // FIXME: SEH catchpads do not create EH scope/funclets, so we could avoid
258 // setting this in such cases in order to improve frame layout.
259 if (!isa<LandingPadInst>(PadInst)) {
260 MF->setHasEHScopes(true);
261 MF->setHasEHFunclets(true);
262 MF->getFrameInfo().setHasOpaqueSPAdjustment(true);
263 }
264 if (isa<CatchSwitchInst>(PadInst)) {
265 assert(BB.begin() == PadInst &&
266 "WinEHPrepare failed to remove PHIs from imaginary BBs");
267 continue;
268 }
269 if (isa<FuncletPadInst>(PadInst) &&
270 Personality != EHPersonality::Wasm_CXX)
271 assert(BB.begin() == PadInst && "WinEHPrepare failed to demote PHIs");
272 }
273
274 MachineBasicBlock *MBB = mf.CreateMachineBasicBlock(&BB);
275 MBBMap[BB.getNumber()] = MBB;
276 MF->push_back(MBB);
277
278 // Transfer the address-taken flag. This is necessary because there could
279 // be multiple MachineBasicBlocks corresponding to one BasicBlock, and only
280 // the first one should be marked.
281 // Only mark the block if the BlockAddress actually has users. The
282 // hasAddressTaken flag may be stale if the BlockAddress was optimized away
283 // but the constant still exists in the uniquing table.
284 if (BB.hasAddressTaken()) {
285 if (BlockAddress *BA = BlockAddress::lookup(&BB))
286 if (!BA->hasZeroLiveUses())
287 MBB->setAddressTakenIRBlock(const_cast<BasicBlock *>(&BB));
288 }
289
290 // Mark landing pad blocks.
291 if (BB.isEHPad())
292 MBB->setIsEHPad();
293
294 // Create Machine PHI nodes for LLVM PHI nodes, lowering them as
295 // appropriate.
296 for (const PHINode &PN : BB.phis()) {
297 if (PN.use_empty())
298 continue;
299
300 // Skip empty types
301 if (PN.getType()->isEmptyTy())
302 continue;
303
304 DebugLoc DL = PN.getDebugLoc();
305 Register PHIReg = ValueMap[&PN];
306 assert(PHIReg && "PHI node does not have an assigned virtual register!");
307
308 SmallVector<EVT, 4> ValueVTs;
309 ComputeValueVTs(*TLI, MF->getDataLayout(), PN.getType(), ValueVTs);
310 for (EVT VT : ValueVTs) {
311 unsigned NumRegisters = TLI->getNumRegisters(Fn->getContext(), VT);
312 const TargetInstrInfo *TII = MF->getSubtarget().getInstrInfo();
313 for (unsigned i = 0; i != NumRegisters; ++i)
314 BuildMI(MBB, DL, TII->get(TargetOpcode::PHI), PHIReg + i);
315 PHIReg += NumRegisters;
316 }
317 }
318 }
319
320 if (isFuncletEHPersonality(Personality)) {
321 WinEHFuncInfo &EHInfo = *MF->getWinEHFuncInfo();
322
323 // Map all BB references in the WinEH data to MBBs.
324 for (WinEHTryBlockMapEntry &TBME : EHInfo.TryBlockMap) {
325 for (WinEHHandlerType &H : TBME.HandlerArray) {
326 if (H.Handler)
327 H.Handler = getMBB(cast<const BasicBlock *>(H.Handler));
328 }
329 }
330 for (CxxUnwindMapEntry &UME : EHInfo.CxxUnwindMap)
331 if (UME.Cleanup)
332 UME.Cleanup = getMBB(cast<const BasicBlock *>(UME.Cleanup));
333 for (SEHUnwindMapEntry &UME : EHInfo.SEHUnwindMap)
334 UME.Handler = getMBB(cast<const BasicBlock *>(UME.Handler));
335 for (ClrEHUnwindMapEntry &CME : EHInfo.ClrEHUnwindMap)
336 CME.Handler = getMBB(cast<const BasicBlock *>(CME.Handler));
337 } else if (Personality == EHPersonality::Wasm_CXX) {
338 WasmEHFuncInfo &EHInfo = *MF->getWasmEHFuncInfo();
339 calculateWasmEHInfo(&fn, EHInfo);
340
341 // Map all BB references in the Wasm EH data to MBBs.
342 DenseMap<BBOrMBB, BBOrMBB> SrcToUnwindDest;
343 for (auto &KV : EHInfo.SrcToUnwindDest) {
344 const auto *Src = cast<const BasicBlock *>(KV.first);
345 const auto *Dest = cast<const BasicBlock *>(KV.second);
346 SrcToUnwindDest[getMBB(Src)] = getMBB(Dest);
347 }
348 EHInfo.SrcToUnwindDest = std::move(SrcToUnwindDest);
349 DenseMap<BBOrMBB, SmallPtrSet<BBOrMBB, 4>> UnwindDestToSrcs;
350 for (auto &KV : EHInfo.UnwindDestToSrcs) {
351 const auto *Dest = cast<const BasicBlock *>(KV.first);
352 MachineBasicBlock *DestMBB = getMBB(Dest);
353 auto &Srcs = UnwindDestToSrcs[DestMBB];
354 for (const auto P : KV.second)
355 Srcs.insert(getMBB(cast<const BasicBlock *>(P)));
356 }
357 EHInfo.UnwindDestToSrcs = std::move(UnwindDestToSrcs);
358 }
359}

387/// CreateRegs - Allocate the appropriate number of virtual registers of
388/// the correctly promoted or expanded types. Assign these registers
389/// consecutive vreg numbers and return the first assigned number.
390///
391/// In the case that the given value has struct or array type, this function
392/// will assign registers for each member or element.
393///
394Register FunctionLoweringInfo::CreateRegs(Type *Ty, bool isDivergent) {
395 SmallVector<EVT, 4> ValueVTs;
396 ComputeValueVTs(*TLI, MF->getDataLayout(), Ty, ValueVTs);
397
398 Register FirstReg;
399 for (EVT ValueVT : ValueVTs) {
400 MVT RegisterVT = TLI->getRegisterType(Ty->getContext(), ValueVT);
401
402 unsigned NumRegs = TLI->getNumRegisters(Ty->getContext(), ValueVT);
403 for (unsigned i = 0; i != NumRegs; ++i) {
404 Register R = CreateReg(RegisterVT, isDivergent);
405 if (!FirstReg) FirstReg = R;
406 }
407 }
408 return FirstReg;
409}
410
411Register FunctionLoweringInfo::CreateRegs(const Value *V) {
412 return CreateRegs(V->getType(), UA && UA->isDivergent(V) &&
413 !TLI->requiresUniformRegister(*MF, V));
414}

416Register FunctionLoweringInfo::InitializeRegForValue(const Value *V) {
417 // Tokens live in vregs only when used for convergence control.
418 if (V->getType()->isTokenTy() && !isa<ConvergenceControlInst>(V))
419 return 0;
420 Register &R = ValueMap[V];
421 assert(R == Register() && "Already initialized this value register!");
422 assert(VirtReg2Value.empty());
423 return R = CreateRegs(V);
424}

426/// GetLiveOutRegInfo - Gets LiveOutInfo for a register, returning NULL if the
427/// register is a PHI destination and the PHI's LiveOutInfo is not valid. If
428/// the register's LiveOutInfo is for a smaller bit width, it is extended to
429/// the larger bit width by zero extension. The bit width must be no smaller
430/// than the LiveOutInfo's existing bit width.
431const FunctionLoweringInfo::LiveOutInfo *
432FunctionLoweringInfo::GetLiveOutRegInfo(Register Reg, unsigned BitWidth) {
433 if (!LiveOutRegInfo.inBounds(Reg))
434 return nullptr;
435
436 LiveOutInfo *LOI = &LiveOutRegInfo[Reg];
437 if (!LOI->IsValid)
438 return nullptr;
439
440 if (BitWidth > LOI->Known.getBitWidth()) {
441 LOI->NumSignBits = 1;
442 LOI->Known = LOI->Known.anyext(BitWidth);
443 }
444
445 return LOI;
446}

448/// ComputePHILiveOutRegInfo - Compute LiveOutInfo for a PHI's destination
449/// register based on the LiveOutInfo of its operands.
450void FunctionLoweringInfo::ComputePHILiveOutRegInfo(const PHINode *PN) {
451 Type *Ty = PN->getType();
452 if (!Ty->isIntegerTy())
453 return;
454
455 SmallVector<EVT, 1> ValueVTs;
456 ComputeValueVTs(*TLI, MF->getDataLayout(), Ty, ValueVTs);
457 assert(ValueVTs.size() == 1 &&
458 "PHIs with non-vector integer types should have a single VT.");
459 EVT IntVT = ValueVTs[0];
460
461 unsigned NumRegisters = TLI->getNumRegisters(PN->getContext(), IntVT);
462 // FIXME: Support multiple registers for big endian targets.
463 if (NumRegisters != 1 && MF->getDataLayout().isBigEndian())
464 return;
465 IntVT = TLI->getRegisterType(PN->getContext(), IntVT);
466 unsigned BitWidth = IntVT.getSizeInBits();
467
468 auto It = ValueMap.find(PN);
469 if (It == ValueMap.end())
470 return;
471
472 Register BaseReg = It->second;
473 if (!BaseReg)
474 return;
475 assert(BaseReg.isVirtual() && "Expected a virtual reg");
476
477 for (unsigned RegIdx = 0; RegIdx < NumRegisters; ++RegIdx) {
478 // Split registers are assigned sequentially.
479 Register DestReg = BaseReg.id() + RegIdx;
480 LiveOutRegInfo.grow(DestReg);
481 LiveOutInfo &DestLOI = LiveOutRegInfo[DestReg];
482
483 Value *V = PN->getIncomingValue(0);
484 if (isa<UndefValue>(V) || isa<ConstantExpr>(V)) {
485 DestLOI.NumSignBits = 1;
486 DestLOI.Known = KnownBits(BitWidth);
487 continue;
488 }
489
490 if (ConstantInt *CI = dyn_cast<ConstantInt>(V)) {
491 APInt Val;
492 if (TLI->signExtendConstant(CI))
493 Val = CI->getValue().sext(BitWidth * NumRegisters);
494 else
495 Val = CI->getValue().zext(BitWidth * NumRegisters);
496 APInt Extracted = Val.extractBits(BitWidth, BitWidth * RegIdx);
497 DestLOI.NumSignBits = Extracted.getNumSignBits();
498 DestLOI.Known = KnownBits::makeConstant(Extracted);
499 } else {
500 assert(ValueMap.count(V) &&
501 "V should have been placed in ValueMap when its"
502 "CopyToReg node was created.");
503 Register SrcReg = ValueMap[V];
504 if (!SrcReg.isVirtual()) {
505 DestLOI.IsValid = false;
506 continue;
507 }
508 // Split registers are assigned sequentially.
509 SrcReg = SrcReg.id() + RegIdx;
510 const LiveOutInfo *SrcLOI = GetLiveOutRegInfo(SrcReg, BitWidth);
511 if (!SrcLOI) {
512 DestLOI.IsValid = false;
513 continue;
514 }
515 DestLOI = *SrcLOI;
516 }
517
518 assert(DestLOI.Known.Zero.getBitWidth() == BitWidth &&
519 DestLOI.Known.One.getBitWidth() == BitWidth &&
520 "Masks should have the same bit width as the type.");
521
522 for (unsigned i = 1, e = PN->getNumIncomingValues(); i != e; ++i) {
523 Value *V = PN->getIncomingValue(i);
524 if (isa<UndefValue>(V) || isa<ConstantExpr>(V)) {
525 DestLOI.NumSignBits = 1;
526 DestLOI.Known = KnownBits(BitWidth);
527 break;
528 }
529
530 if (ConstantInt *CI = dyn_cast<ConstantInt>(V)) {
531 APInt Val;
532 if (TLI->signExtendConstant(CI))
533 Val = CI->getValue().sext(BitWidth * NumRegisters);
534 else
535 Val = CI->getValue().zext(BitWidth * NumRegisters);
536 APInt Extracted = Val.extractBits(BitWidth, BitWidth * RegIdx);
537 DestLOI.NumSignBits =
538 std::min(DestLOI.NumSignBits, Extracted.getNumSignBits());
539 DestLOI.Known =
540 DestLOI.Known.intersectWith(KnownBits::makeConstant(Extracted));
541 continue;
542 }
543
544 assert(ValueMap.count(V) && "V should have been placed in ValueMap when "
545 "its CopyToReg node was created.");
546 Register SrcReg = ValueMap[V];
547 if (!SrcReg.isVirtual()) {
548 DestLOI.IsValid = false;
549 break;
550 }
551 // Split registers are assigned sequentially.
552 SrcReg = SrcReg.id() + RegIdx;
553 const LiveOutInfo *SrcLOI = GetLiveOutRegInfo(SrcReg, BitWidth);
554 if (!SrcLOI) {
555 DestLOI.IsValid = false;
556 break;
557 }
558 DestLOI.NumSignBits = std::min(DestLOI.NumSignBits, SrcLOI->NumSignBits);
559 DestLOI.Known = DestLOI.Known.intersectWith(SrcLOI->Known);
560 }
561 }

594const Value *
595FunctionLoweringInfo::getValueFromVirtualReg(Register Vreg) {
596 if (VirtReg2Value.empty()) {
597 SmallVector<EVT, 4> ValueVTs;
598 for (auto &P : ValueMap) {
599 ValueVTs.clear();
600 ComputeValueVTs(*TLI, Fn->getDataLayout(),
601 P.first->getType(), ValueVTs);
602 Register Reg = P.second;
603 for (EVT VT : ValueVTs) {
604 unsigned NumRegisters = TLI->getNumRegisters(Fn->getContext(), VT);
605 for (unsigned i = 0, e = NumRegisters; i != e; ++i)
606 VirtReg2Value[Reg++] = P.first;
607 }
608 }
609 }
610 return VirtReg2Value.lookup(Vreg);
611}
