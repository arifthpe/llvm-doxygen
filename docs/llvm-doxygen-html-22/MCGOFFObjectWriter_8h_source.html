49// A GOFFRelocationEntry describes a single relocation.
50// For the naming, see
51// https://www.ibm.com/docs/en/zos/3.1.0?topic=record-relocation-directory-data-item.
52struct GOFFRelocationEntry {
53 const MCSymbolGOFF *Rptr; // The R pointer.
54 const MCSectionGOFF *Pptr; // The P pointer.
55 uint32_t REsdId = 0; // The R pointer id.
56 uint32_t PEsdId = 0; // The P pointer id.
57 uint64_t POffset; // The offset within the element described by the P pointer.
58 uint32_t TargetLength; // The byte length of the target field.
59
60 // Details of the relocation.
61 GOFF::RLDReferenceType ReferenceType : 4;
62 GOFF::RLDReferentType ReferentType : 2;
63 GOFF::RLDAction Action : 1;
64 GOFF::RLDFetchStore FetchStore : 1;
65
66 GOFFRelocationEntry(const MCSectionGOFF *Pptr, const MCSymbolGOFF *Rptr,
67 GOFF::RLDReferenceType ReferenceType,
68 GOFF::RLDReferentType ReferentType,
69 GOFF::RLDAction Action, GOFF::RLDFetchStore FetchStore,
70 uint64_t POffset, uint32_t TargetLength)
71 : Rptr(Rptr), Pptr(Pptr), POffset(POffset), TargetLength(TargetLength),
72 ReferenceType(ReferenceType), ReferentType(ReferentType),
73 Action(Action), FetchStore(FetchStore) {}
74};
