43PreservedAnalyses
44MachineBranchProbabilityPrinterPass::run(MachineFunction &MF,
45 MachineFunctionAnalysisManager &MFAM) {
46 OS << "Printing analysis 'Machine Branch Probability Analysis' for machine "
47 "function '"
48 << MF.getName() << "':\n";
49 auto &MBPI = MFAM.getResult<MachineBranchProbabilityAnalysis>(MF);
50 for (const MachineBasicBlock &MBB : MF) {
51 for (const MachineBasicBlock *Succ : MBB.successors())
52 MBPI.printEdgeProbability(OS << " ", &MBB, Succ);
53 }
54 return PreservedAnalyses::all();
55}

83BranchProbability MachineBranchProbabilityInfo::getEdgeProbability(
84 const MachineBasicBlock *Src, const MachineBasicBlock *Dst) const {
85 // This is a linear search. Try to use the const_succ_iterator version when
86 // possible.
87 return getEdgeProbability(Src, find(Src->successors(), Dst));
88}

90bool MachineBranchProbabilityInfo::isEdgeHot(
91 const MachineBasicBlock *Src, const MachineBasicBlock *Dst) const {
92 BranchProbability HotProb(StaticLikelyProb, 100);
93 return getEdgeProbability(Src, Dst) > HotProb;
94}
95
96raw_ostream &MachineBranchProbabilityInfo::printEdgeProbability(
97 raw_ostream &OS, const MachineBasicBlock *Src,
98 const MachineBasicBlock *Dst) const {
99
100 const BranchProbability Prob = getEdgeProbability(Src, Dst);
101 OS << "edge " << printMBBReference(*Src) << " -> " << printMBBReference(*Dst)
102 << " probability is " << Prob
103 << (isEdgeHot(Src, Dst) ? " [HOT edge]\n" : "\n");
104
105 return OS;
106}
