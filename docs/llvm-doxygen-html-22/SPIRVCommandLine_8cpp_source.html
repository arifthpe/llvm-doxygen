183bool SPIRVExtensionsParser::parse(cl::Option &O, StringRef ArgName,
184 StringRef ArgValue,
185 std::set<SPIRV::Extension::Extension> &Vals) {
186 SmallVector<StringRef, 10> Tokens;
187 ArgValue.split(Tokens, ",", -1, false);
188
189 std::set<SPIRV::Extension::Extension> EnabledExtensions;
190
191 auto M = partition(Tokens, [](auto &&T) { return T.starts_with('+'); });
192
193 if (std::any_of(M, Tokens.end(), [](auto &&T) { return T == "all"; }))
194 copy(make_second_range(SPIRVExtensionMap), std::inserter(Vals, Vals.end()));
195
196 for (auto &&Token : make_range(Tokens.begin(), M)) {
197 StringRef ExtensionName = Token.substr(1);
198 auto NameValuePair = SPIRVExtensionMap.find(ExtensionName);
199
200 if (NameValuePair == SPIRVExtensionMap.end())
201 return O.error("Unknown SPIR-V extension: " + Token.str());
202
203 EnabledExtensions.insert(NameValuePair->second);
204 }
205
206 for (auto &&Token : make_range(M, Tokens.end())) {
207 if (Token == "all")
208 continue;
209
210 if (Token.size() == 3 && Token.upper() == "KHR") {
211 for (const auto &[ExtensionName, ExtensionEnum] : SPIRVExtensionMap)
212 if (StringRef(ExtensionName).starts_with("SPV_KHR_"))
213 Vals.insert(ExtensionEnum);
214 continue;
215 }
216
217 if (Token.empty() || (!Token.starts_with("+") && !Token.starts_with("-")))
218 return O.error("Invalid extension list format: " + Token);
219
220 auto NameValuePair = SPIRVExtensionMap.find(Token.substr(1));
221
222 if (NameValuePair == SPIRVExtensionMap.cend())
223 return O.error("Unknown SPIR-V extension: " + Token.str());
224 if (EnabledExtensions.count(NameValuePair->second))
225 return O.error(
226 "Extension cannot be allowed and disallowed at the same time: " +
227 NameValuePair->first);
228
229 Vals.erase(NameValuePair->second);
230 }
231
232 Vals.insert(EnabledExtensions.cbegin(), EnabledExtensions.cend());
233
234 return false;
235}

237StringRef SPIRVExtensionsParser::checkExtensions(
238 const std::vector<std::string> &ExtNames,
239 std::set<SPIRV::Extension::Extension> &AllowedExtensions) {
240 for (const auto &Ext : ExtNames) {
241 if (Ext == "all") {
242 for (const auto &[ExtensionName, ExtensionEnum] : SPIRVExtensionMap)
243 AllowedExtensions.insert(ExtensionEnum);
244 break;
245 }
246 auto It = SPIRVExtensionMap.find(Ext);
247 if (It == SPIRVExtensionMap.end())
248 return Ext;
249 AllowedExtensions.insert(It->second);
250 }
251 return StringRef();
252}

254std::set<SPIRV::Extension::Extension>
255SPIRVExtensionsParser::getValidExtensions(const Triple &TT) {
256 std::set<SPIRV::Extension::Extension> R;
257 SPIRV::Environment::Environment CurrentEnvironment =
258 SPIRV::Environment::Environment::EnvOpenCL;
259 if (TT.getOS() == Triple::Vulkan)
260 CurrentEnvironment = SPIRV::Environment::Environment::EnvVulkan;
261
262 for (const auto &[ExtensionName, ExtensionEnum] : SPIRVExtensionMap) {
263 EnvironmentList AllowedEnv = getSymbolicOperandAllowedEnvironments(
264 SPIRV::OperandCategory::OperandCategory::ExtensionOperand,
265 ExtensionEnum);
266
267 if (llvm::is_contained(AllowedEnv, CurrentEnvironment))
268 R.insert(ExtensionEnum);
269 }
270
271 if (TT.getVendor() == Triple::AMD) {
272 // AMD uses the translator to recover LLVM-IR from SPIRV. Currently, the
273 // translator doesn't implement the SPV_KHR_float_controls2 extension.
274 R.erase(SPIRV::Extension::SPV_KHR_float_controls2);
275 }
276
277 return R;
278}
