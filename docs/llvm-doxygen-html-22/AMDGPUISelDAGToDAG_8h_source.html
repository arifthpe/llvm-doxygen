28static inline bool getConstantValue(SDValue N, uint32_t &Out) {
29 // This is only used for packed vectors, where using 0 for undef should
30 // always be good.
31 if (N.isUndef()) {
32 Out = 0;
33 return true;
34 }
35
36 if (const ConstantSDNode *C = dyn_cast<ConstantSDNode>(N)) {
37 Out = C->getAPIntValue().getSExtValue();
38 return true;
39 }
40
41 if (const ConstantFPSDNode *C = dyn_cast<ConstantFPSDNode>(N)) {
42 Out = C->getValueAPF().bitcastToAPInt().getSExtValue();
43 return true;
44 }
45
46 return false;
47}
