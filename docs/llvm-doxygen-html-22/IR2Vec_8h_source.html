8///
9/// \file
10/// This file defines the IR2Vec vocabulary analysis(IR2VecVocabAnalysis),
11/// the core ir2vec::Embedder interface for generating IR embeddings,
12/// and related utilities like the IR2VecPrinterPass.
13///
14/// Program Embeddings are typically or derived-from a learned
15/// representation of the program. Such embeddings are used to represent the
16/// programs as input to machine learning algorithms. IR2Vec represents the
17/// LLVM IR as embeddings.
18///
19/// The IR2Vec algorithm is described in the following paper:
20///
21/// IR2Vec: LLVM IR Based Scalable Program Embeddings, S. VenkataKeerthy,
22/// Rohit Aggarwal, Shalini Jain, Maunendra Sankar Desarkar, Ramakrishna
23/// Upadrasta, and Y. N. Srikant, ACM Transactions on Architecture and
24/// Code Optimization (TACO), 2020. https://doi.org/10.1145/3418463.
25/// https://arxiv.org/abs/1909.06228
26///
27/// To obtain embeddings:
28/// First run IR2VecVocabAnalysis to populate the vocabulary.
29/// Then, use the Embedder interface to generate embeddings for the desired IR
30/// entities. See the documentation for more details -
31/// https://llvm.org/docs/MLGO.html#ir2vec-embeddings
32///
33//===----------------------------------------------------------------------===//

61/// IR2Vec computes two kinds of embeddings: Symbolic and Flow-aware.
62/// Symbolic embeddings capture the "syntactic" and "statistical correlation"
63/// of the IR entities. Flow-aware embeddings build on top of symbolic
64/// embeddings and additionally capture the flow information in the IR.
65/// IR2VecKind is used to specify the type of embeddings to generate.
66/// Note: Implementation of FlowAware embeddings is not same as the one
67/// described in the paper. The current implementation is a simplified version
68/// that captures the flow information (SSA-based use-defs) without tracing
69/// through memory level use-defs in the embedding computation described in the
70/// paper.
71enum class IR2VecKind { Symbolic, FlowAware };

81/// Embedding is a datatype that wraps std::vector<double>. It provides
82/// additional functionality for arithmetic and comparison operations.
83/// It is meant to be used *like* std::vector<double> but is more restrictive
84/// in the sense that it does not allow the user to change the size of the
85/// embedding vector. The dimension of the embedding is fixed at the time of
86/// construction of Embedding object. But the elements can be modified in-place.
87struct Embedding {
88private:
89 std::vector<double> Data;
90
91public:
92 Embedding() = default;
93 Embedding(const std::vector<double> &V) : Data(V) {}
94 Embedding(std::vector<double> &&V) : Data(std::move(V)) {}
95 Embedding(std::initializer_list<double> IL) : Data(IL) {}
96
97 explicit Embedding(size_t Size) : Data(Size, 0.0) {}
98 Embedding(size_t Size, double InitialValue) : Data(Size, InitialValue) {}
99
100 size_t size() const { return Data.size(); }
101 bool empty() const { return Data.empty(); }
102
103 double &operator[](size_t Itr) {
104 assert(Itr < Data.size() && "Index out of bounds");
105 return Data[Itr];
106 }
107
108 const double &operator[](size_t Itr) const {
109 assert(Itr < Data.size() && "Index out of bounds");
110 return Data[Itr];
111 }
112
113 using iterator = std::vector<double>::iterator;
114 using const_iterator = std::vector<double>::const_iterator;
115
116 iterator begin() { return Data.begin(); }
117 iterator end() { return Data.end(); }
118 const_iterator begin() const { return Data.begin(); }
119 const_iterator end() const { return Data.end(); }
120 const_iterator cbegin() const { return Data.cbegin(); }
121 const_iterator cend() const { return Data.cend(); }
122
123 const std::vector<double> &getData() const { return Data; }
124
125 /// Arithmetic operators
126 LLVM_ABI Embedding &operator+=(const Embedding &RHS);
127 LLVM_ABI Embedding operator+(const Embedding &RHS) const;
128 LLVM_ABI Embedding &operator-=(const Embedding &RHS);
129 LLVM_ABI Embedding operator-(const Embedding &RHS) const;
130 LLVM_ABI Embedding &operator*=(double Factor);
131 LLVM_ABI Embedding operator*(double Factor) const;
132
133 /// Adds Src Embedding scaled by Factor with the called Embedding.
134 /// Called_Embedding += Src * Factor
135 LLVM_ABI Embedding &scaleAndAdd(const Embedding &Src, float Factor);
136
137 /// Returns true if the embedding is approximately equal to the RHS embedding
138 /// within the specified tolerance.
139 LLVM_ABI bool approximatelyEquals(const Embedding &RHS,
140 double Tolerance = 1e-4) const;
141
142 LLVM_ABI void print(raw_ostream &OS) const;
143};

225/// Class for storing and accessing the IR2Vec vocabulary.
226/// The Vocabulary class manages seed embeddings for LLVM IR entities. The
227/// seed embeddings are the initial learned representations of the entities
228/// of LLVM IR. The IR2Vec representation for a given IR is derived from these
229/// seed embeddings.
230///
231/// The vocabulary contains the seed embeddings for three types of entities:
232/// instruction opcodes, types, and operands. Types are grouped/canonicalized
233/// for better learning (e.g., all float variants map to FloatTy). The
234/// vocabulary abstracts away the canonicalization effectively, the exposed APIs
235/// handle all the known LLVM IR opcodes, types and operands.
236///
237/// This class helps populate the seed embeddings in an internal vector-based
238/// ADT. It provides logic to map every IR entity to a specific slot index or
239/// position in this vector, enabling O(1) embedding lookup while avoiding
240/// unnecessary computations involving string based lookups while generating the
241/// embeddings.
242class Vocabulary {
243 friend class llvm::IR2VecVocabAnalysis;
244
245 // Vocabulary Layout:
246 // +----------------+------------------------------------------------------+
247 // | Entity Type | Index Range |
248 // +----------------+------------------------------------------------------+
249 // | Opcodes | [0 .. (MaxOpcodes-1)] |
250 // | Canonical Types| [MaxOpcodes .. (MaxOpcodes+MaxCanonicalTypeIDs-1)] |
251 // | Operands | [(MaxOpcodes+MaxCanonicalTypeIDs) .. NumCanEntries] |
252 // +----------------+------------------------------------------------------+
253 // Note: MaxOpcodes is the number of unique opcodes supported by LLVM IR.
254 // MaxCanonicalTypeIDs is the number of canonicalized type IDs.
255 // "Similar" LLVM Types are grouped/canonicalized together. E.g., all
256 // float variants (FloatTy, DoubleTy, HalfTy, etc.) map to
257 // CanonicalTypeID::FloatTy. This helps reduce the vocabulary size
258 // and improves learning. Operands include Comparison predicates
259 // (ICmp/FCmp) along with other operand types. This can be extended to
260 // include other specializations in future.
261 enum class Section : unsigned {
262 Opcodes = 0,
263 CanonicalTypes = 1,
264 Operands = 2,
265 Predicates = 3,
266 MaxSections
267 };

518/// Embedder provides the interface to generate embeddings (vector
519/// representations) for instructions, basic blocks, and functions. The
520/// vector representations are generated using IR2Vec algorithms.
521///
522/// The Embedder class is an abstract class and it is intended to be
523/// subclassed for different IR2Vec algorithms like Symbolic and Flow-aware.
524class Embedder {
525protected:
526 const Function &F;
527 const Vocabulary &Vocab;
528
529 /// Dimension of the vector representation; captured from the input vocabulary
530 const unsigned Dimension;
531
532 /// Weights for different entities (like opcode, arguments, types)
533 /// in the IR instructions to generate the vector representation.
534 const float OpcWeight, TypeWeight, ArgWeight;
535
536 LLVM_ABI Embedder(const Function &F, const Vocabulary &Vocab)
537 : F(F), Vocab(Vocab), Dimension(Vocab.getDimension()),
538 OpcWeight(ir2vec::OpcWeight), TypeWeight(ir2vec::TypeWeight),
539 ArgWeight(ir2vec::ArgWeight) {}
540
541 /// Function to compute embeddings.
542 Embedding computeEmbeddings() const;
543
544 /// Function to compute the embedding for a given basic block.
545 Embedding computeEmbeddings(const BasicBlock &BB) const;
546
547 /// Function to compute the embedding for a given instruction.
548 /// Specific to the kind of embeddings being computed.
549 virtual Embedding computeEmbeddings(const Instruction &I) const = 0;
550
551public:
552 virtual ~Embedder() = default;
553
554 /// Factory method to create an Embedder object.
555 LLVM_ABI static std::unique_ptr<Embedder>
556 create(IR2VecKind Mode, const Function &F, const Vocabulary &Vocab);
557
558 /// Computes and returns the embedding for a given instruction in the function
559 /// F
560 LLVM_ABI Embedding getInstVector(const Instruction &I) const {
561 return computeEmbeddings(I);
562 }
563
564 /// Computes and returns the embedding for a given basic block in the function
565 /// F
566 LLVM_ABI Embedding getBBVector(const BasicBlock &BB) const {
567 return computeEmbeddings(BB);
568 }
569
570 /// Computes and returns the embedding for the current function.
571 LLVM_ABI Embedding getFunctionVector() const { return computeEmbeddings(); }
572
573 /// Invalidate embeddings if cached. The embeddings may not be relevant
574 /// anymore when the IR changes due to transformations. In such cases, the
575 /// cached embeddings should be invalidated to ensure
576 /// correctness/recomputation. This is a no-op for SymbolicEmbedder but
577 /// removes all the cached entries in FlowAwareEmbedder.
578 virtual void invalidateEmbeddings() {}
579};
