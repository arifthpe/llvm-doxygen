32class RegisterScheduler
33 : public MachinePassRegistryNode<ScheduleDAGSDNodes *(*)(SelectionDAGISel *,
34 CodeGenOptLevel)> {
35public:
36 using FunctionPassCtor = ScheduleDAGSDNodes *(*)(SelectionDAGISel *,
37 CodeGenOptLevel);
38
39 LLVM_ABI static MachinePassRegistry<FunctionPassCtor> Registry;
40
41 RegisterScheduler(const char *N, const char *D, FunctionPassCtor C)
42 : MachinePassRegistryNode(N, D, C) {
43 Registry.Add(this);
44 }
45 ~RegisterScheduler() { Registry.Remove(this); }
46
47
48 // Accessors.

50 return (RegisterScheduler *)MachinePassRegistryNode::getNext();
51 }
52
53 static RegisterScheduler *getList() {
54 return (RegisterScheduler *)Registry.getList();
55 }
56
57 static void setListener(MachinePassRegistryListener<FunctionPassCtor> *L) {
58 Registry.setListener(L);
59 }
60};
