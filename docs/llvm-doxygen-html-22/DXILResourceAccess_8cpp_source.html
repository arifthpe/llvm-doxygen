31static Value *traverseGEPOffsets(const DataLayout &DL, IRBuilder<> &Builder,
32 Value *Ptr, uint64_t AccessSize) {
33 Value *Offset = nullptr;
34
35 while (Ptr) {
36 if (auto *II = dyn_cast<IntrinsicInst>(Ptr)) {
37 assert(II->getIntrinsicID() == Intrinsic::dx_resource_getpointer &&
38 "Resource access through unexpected intrinsic");
39 return Offset ? Offset : ConstantInt::get(Builder.getInt32Ty(), 0);
40 }
41
42 auto *GEP = dyn_cast<GetElementPtrInst>(Ptr);
43 assert(GEP && "Resource access through unexpected instruction");
44
45 unsigned NumIndices = GEP->getNumIndices();
46 uint64_t IndexScale = DL.getTypeAllocSize(GEP->getSourceElementType());
47 APInt ConstantOffset(DL.getIndexTypeSizeInBits(GEP->getType()), 0);
48 Value *GEPOffset;
49 if (GEP->accumulateConstantOffset(DL, ConstantOffset)) {
50 // We have a constant offset (in bytes).
51 GEPOffset =
52 ConstantInt::get(DL.getIndexType(GEP->getType()), ConstantOffset);
53 IndexScale = 1;
54 } else if (NumIndices == 1) {
55 // If we have a single index we're indexing into a top level array. This
56 // generally only happens with cbuffers.
57 GEPOffset = *GEP->idx_begin();
58 } else if (NumIndices == 2) {
59 // If we have two indices, this should be an access through a pointer.
60 auto IndexIt = GEP->idx_begin();
61 assert(cast<ConstantInt>(IndexIt)->getZExtValue() == 0 &&
62 "GEP is not indexing through pointer");
63 GEPOffset = *(++IndexIt);
64 } else
65 llvm_unreachable("Unhandled GEP structure for resource access");
66
67 uint64_t ElemSize = AccessSize;
68 if (!(IndexScale % ElemSize)) {
69 // If our scale is an exact multiple of the access size, adjust the
70 // scaling to avoid an unnecessary division.
71 IndexScale /= ElemSize;
72 ElemSize = 1;
73 }
74 if (IndexScale != 1)
75 GEPOffset = Builder.CreateMul(
76 GEPOffset, ConstantInt::get(Builder.getInt32Ty(), IndexScale));
77 if (ElemSize != 1)
78 GEPOffset = Builder.CreateUDiv(
79 GEPOffset, ConstantInt::get(Builder.getInt32Ty(), ElemSize));
80
81 Offset = Offset ? Builder.CreateAdd(Offset, GEPOffset) : GEPOffset;
82 Ptr = GEP->getPointerOperand();
83 }
84
85 llvm_unreachable("GEP of null pointer?");
86}

88static void createTypedBufferStore(IntrinsicInst *II, StoreInst *SI,
89 dxil::ResourceTypeInfo &RTI) {
90 const DataLayout &DL = SI->getDataLayout();
91 IRBuilder<> Builder(SI);
92 Type *ContainedType = RTI.getHandleTy()->getTypeParameter(0);
93 Type *ScalarType = ContainedType->getScalarType();
94 Type *LoadType = StructType::get(ContainedType, Builder.getInt1Ty());
95
96 Value *V = SI->getValueOperand();
97 if (V->getType() == ContainedType) {
98 // V is already the right type.
99 assert(SI->getPointerOperand() == II &&
100 "Store of whole element has mismatched address to store to");
101 } else if (V->getType() == ScalarType) {
102 // We're storing a scalar, so we need to load the current value and only
103 // replace the relevant part.
104 auto *Load = Builder.CreateIntrinsic(
105 LoadType, Intrinsic::dx_resource_load_typedbuffer,
106 {II->getOperand(0), II->getOperand(1)});
107 auto *Struct = Builder.CreateExtractValue(Load, {0});
108
109 uint64_t AccessSize = DL.getTypeSizeInBits(ScalarType) / 8;
110 Value *Offset =
111 traverseGEPOffsets(DL, Builder, SI->getPointerOperand(), AccessSize);
112 V = Builder.CreateInsertElement(Struct, V, Offset);
113 } else {
114 llvm_unreachable("Store to typed resource has invalid type");
115 }
116
117 auto *Inst = Builder.CreateIntrinsic(
118 Builder.getVoidTy(), Intrinsic::dx_resource_store_typedbuffer,
119 {II->getOperand(0), II->getOperand(1), V});
120 SI->replaceAllUsesWith(Inst);
121}

123static void createRawStore(IntrinsicInst *II, StoreInst *SI,
124 dxil::ResourceTypeInfo &RTI) {
125 const DataLayout &DL = SI->getDataLayout();
126 IRBuilder<> Builder(SI);
127
128 Value *V = SI->getValueOperand();
129 assert(!V->getType()->isAggregateType() &&
130 "Resource store should be scalar or vector type");
131
132 Value *Index = II->getOperand(1);
133 // The offset for the rawbuffer load and store ops is always in bytes.
134 uint64_t AccessSize = 1;
135 Value *Offset =
136 traverseGEPOffsets(DL, Builder, SI->getPointerOperand(), AccessSize);
137
138 // For raw buffer (ie, HLSL's ByteAddressBuffer), we need to fold the access
139 // entirely into the index.
140 if (!RTI.isStruct()) {
141 auto *ConstantOffset = dyn_cast<ConstantInt>(Offset);
142 if (!ConstantOffset || !ConstantOffset->isZero())
143 Index = Builder.CreateAdd(Index, Offset);
144 Offset = llvm::PoisonValue::get(Builder.getInt32Ty());
145 }
146
147 auto *Inst = Builder.CreateIntrinsic(Builder.getVoidTy(),
148 Intrinsic::dx_resource_store_rawbuffer,
149 {II->getOperand(0), Index, Offset, V});
150 SI->replaceAllUsesWith(Inst);
151}

153static void createStoreIntrinsic(IntrinsicInst *II, StoreInst *SI,
154 dxil::ResourceTypeInfo &RTI) {
155 switch (RTI.getResourceKind()) {
156 case dxil::ResourceKind::TypedBuffer:
157 return createTypedBufferStore(II, SI, RTI);
158 case dxil::ResourceKind::RawBuffer:
159 case dxil::ResourceKind::StructuredBuffer:
160 return createRawStore(II, SI, RTI);
161 case dxil::ResourceKind::Texture1D:
162 case dxil::ResourceKind::Texture2D:
163 case dxil::ResourceKind::Texture2DMS:
164 case dxil::ResourceKind::Texture3D:
165 case dxil::ResourceKind::TextureCube:
166 case dxil::ResourceKind::Texture1DArray:
167 case dxil::ResourceKind::Texture2DArray:
168 case dxil::ResourceKind::Texture2DMSArray:
169 case dxil::ResourceKind::TextureCubeArray:
170 case dxil::ResourceKind::FeedbackTexture2D:
171 case dxil::ResourceKind::FeedbackTexture2DArray:
172 reportFatalUsageError("DXIL Load not implemented yet");
173 return;
174 case dxil::ResourceKind::CBuffer:
175 case dxil::ResourceKind::Sampler:
176 case dxil::ResourceKind::TBuffer:
177 case dxil::ResourceKind::RTAccelerationStructure:
178 case dxil::ResourceKind::Invalid:
179 case dxil::ResourceKind::NumEntries:
180 llvm_unreachable("Invalid resource kind for store");
181 }
182 llvm_unreachable("Unhandled case in switch");
183}

185static void createTypedBufferLoad(IntrinsicInst *II, LoadInst *LI,
186 dxil::ResourceTypeInfo &RTI) {
187 const DataLayout &DL = LI->getDataLayout();
188 IRBuilder<> Builder(LI);
189 Type *ContainedType = RTI.getHandleTy()->getTypeParameter(0);
190 Type *LoadType = StructType::get(ContainedType, Builder.getInt1Ty());
191
192 Value *V =
193 Builder.CreateIntrinsic(LoadType, Intrinsic::dx_resource_load_typedbuffer,
194 {II->getOperand(0), II->getOperand(1)});
195 V = Builder.CreateExtractValue(V, {0});
196
197 Type *ScalarType = ContainedType->getScalarType();
198 uint64_t AccessSize = DL.getTypeSizeInBits(ScalarType) / 8;
199 Value *Offset =
200 traverseGEPOffsets(DL, Builder, LI->getPointerOperand(), AccessSize);
201 auto *ConstantOffset = dyn_cast<ConstantInt>(Offset);
202 if (!ConstantOffset || !ConstantOffset->isZero())
203 V = Builder.CreateExtractElement(V, Offset);
204
205 // If we loaded a <1 x ...> instead of a scalar (presumably to feed a
206 // shufflevector), then make sure we're maintaining the resulting type.
207 if (auto *VT = dyn_cast<FixedVectorType>(LI->getType()))
208 if (VT->getNumElements() == 1 && !isa<FixedVectorType>(V->getType()))
209 V = Builder.CreateInsertElement(PoisonValue::get(VT), V,
210 Builder.getInt32(0));
211
212 LI->replaceAllUsesWith(V);
213}

215static void createRawLoad(IntrinsicInst *II, LoadInst *LI,
216 dxil::ResourceTypeInfo &RTI) {
217 const DataLayout &DL = LI->getDataLayout();
218 IRBuilder<> Builder(LI);
219
220 Type *LoadType = StructType::get(LI->getType(), Builder.getInt1Ty());
221 assert(!LI->getType()->isAggregateType() &&
222 "Resource load should be scalar or vector type");
223
224 Value *Index = II->getOperand(1);
225 // The offset for the rawbuffer load and store ops is always in bytes.
226 uint64_t AccessSize = 1;
227 Value *Offset =
228 traverseGEPOffsets(DL, Builder, LI->getPointerOperand(), AccessSize);
229
230 // For raw buffer (ie, HLSL's ByteAddressBuffer), we need to fold the access
231 // entirely into the index.
232 if (!RTI.isStruct()) {
233 auto *ConstantOffset = dyn_cast<ConstantInt>(Offset);
234 if (!ConstantOffset || !ConstantOffset->isZero())
235 Index = Builder.CreateAdd(Index, Offset);
236 Offset = llvm::PoisonValue::get(Builder.getInt32Ty());
237 }
238
239 Value *V =
240 Builder.CreateIntrinsic(LoadType, Intrinsic::dx_resource_load_rawbuffer,
241 {II->getOperand(0), Index, Offset});
242 V = Builder.CreateExtractValue(V, {0});
243
244 LI->replaceAllUsesWith(V);
245}

248/// Helper for building a `load.cbufferrow` intrinsic given a simple type.
249struct CBufferRowIntrin {
250 Intrinsic::ID IID;
251 Type *RetTy;
252 unsigned int EltSize;
253 unsigned int NumElts;
254
255 CBufferRowIntrin(const DataLayout &DL, Type *Ty) {
256 assert(Ty == Ty->getScalarType() && "Expected scalar type");
257
258 switch (DL.getTypeSizeInBits(Ty)) {
259 case 16:
260 IID = Intrinsic::dx_resource_load_cbufferrow_8;
261 RetTy = StructType::get(Ty, Ty, Ty, Ty, Ty, Ty, Ty, Ty);
262 EltSize = 2;
263 NumElts = 8;
264 break;
265 case 32:
266 IID = Intrinsic::dx_resource_load_cbufferrow_4;
267 RetTy = StructType::get(Ty, Ty, Ty, Ty);
268 EltSize = 4;
269 NumElts = 4;
270 break;
271 case 64:
272 IID = Intrinsic::dx_resource_load_cbufferrow_2;
273 RetTy = StructType::get(Ty, Ty);
274 EltSize = 8;
275 NumElts = 2;
276 break;
277 default:
278 llvm_unreachable("Only 16, 32, and 64 bit types supported");
279 }
280 }
281};
282} // namespace

284static void createCBufferLoad(IntrinsicInst *II, LoadInst *LI,
285 dxil::ResourceTypeInfo &RTI) {
286 const DataLayout &DL = LI->getDataLayout();
287
288 Type *Ty = LI->getType();
289 assert(!isa<StructType>(Ty) && "Structs not handled yet");
290 CBufferRowIntrin Intrin(DL, Ty->getScalarType());
291
292 StringRef Name = LI->getName();
293 Value *Handle = II->getOperand(0);
294
295 IRBuilder<> Builder(LI);
296
297 ConstantInt *GlobalOffset = dyn_cast<ConstantInt>(II->getOperand(1));
298 assert(GlobalOffset && "CBuffer getpointer index must be constant");
299
300 uint64_t GlobalOffsetVal = GlobalOffset->getZExtValue();
301 Value *CurrentRow = ConstantInt::get(
302 Builder.getInt32Ty(), GlobalOffsetVal / hlsl::CBufferRowSizeInBytes);
303 unsigned int CurrentIndex =
304 (GlobalOffsetVal % hlsl::CBufferRowSizeInBytes) / Intrin.EltSize;
305
306 // Every object in a cbuffer either fits in a row or is aligned to a row. This
307 // means that only the very last pointer access can point into a row.
308 auto *LastGEP = dyn_cast<GEPOperator>(LI->getPointerOperand());
309 if (!LastGEP) {
310 // If we don't have a GEP at all we're just accessing the resource through
311 // the result of getpointer directly.
312 assert(LI->getPointerOperand() == II &&
313 "Unexpected indirect access to resource without GEP");
314 } else {
315 Value *GEPOffset = traverseGEPOffsets(
316 DL, Builder, LastGEP->getPointerOperand(), hlsl::CBufferRowSizeInBytes);
317 CurrentRow = Builder.CreateAdd(GEPOffset, CurrentRow);
318
319 APInt ConstantOffset(DL.getIndexTypeSizeInBits(LastGEP->getType()), 0);
320 if (LastGEP->accumulateConstantOffset(DL, ConstantOffset)) {
321 APInt Remainder(DL.getIndexTypeSizeInBits(LastGEP->getType()),
322 hlsl::CBufferRowSizeInBytes);
323 APInt::udivrem(ConstantOffset, Remainder, ConstantOffset, Remainder);
324 CurrentRow = Builder.CreateAdd(
325 CurrentRow, ConstantInt::get(Builder.getInt32Ty(), ConstantOffset));
326 CurrentIndex += Remainder.udiv(Intrin.EltSize).getZExtValue();
327 } else {
328 assert(LastGEP->getNumIndices() == 1 &&
329 "Last GEP of cbuffer access is not array or struct access");
330 // We assume a non-constant access will be row-aligned. This is safe
331 // because arrays and structs are always row aligned, and accesses to
332 // vector elements will show up as a load of the vector followed by an
333 // extractelement.
334 CurrentRow = cast<ConstantInt>(CurrentRow)->isZero()
335 ? *LastGEP->idx_begin()
336 : Builder.CreateAdd(CurrentRow, *LastGEP->idx_begin());
337 CurrentIndex = 0;
338 }
339 }
340
341 auto *CBufLoad = Builder.CreateIntrinsic(
342 Intrin.RetTy, Intrin.IID, {Handle, CurrentRow}, nullptr, Name + ".load");
343 auto *Elt =
344 Builder.CreateExtractValue(CBufLoad, {CurrentIndex++}, Name + ".extract");
345
346 // At this point we've loaded the first scalar of our result, but our original
347 // type may have been a vector.
348 unsigned int Remaining =
349 ((DL.getTypeSizeInBits(Ty) / 8) / Intrin.EltSize) - 1;
350 if (Remaining == 0) {
351 // We only have a single element, so we're done.
352 Value *Result = Elt;
353
354 // However, if we loaded a <1 x T>, then we need to adjust the type.
355 if (auto *VT = dyn_cast<FixedVectorType>(Ty)) {
356 assert(VT->getNumElements() == 1 && "Can't have multiple elements here");
357 Result = Builder.CreateInsertElement(PoisonValue::get(VT), Result,
358 Builder.getInt32(0), Name);
359 }
360 LI->replaceAllUsesWith(Result);
361 return;
362 }
363
364 // Walk each element and extract it, wrapping to new rows as needed.
365 SmallVector<Value *> Extracts{Elt};
366 while (Remaining--) {
367 CurrentIndex %= Intrin.NumElts;
368
369 if (CurrentIndex == 0) {
370 CurrentRow = Builder.CreateAdd(CurrentRow,
371 ConstantInt::get(Builder.getInt32Ty(), 1));
372 CBufLoad = Builder.CreateIntrinsic(Intrin.RetTy, Intrin.IID,
373 {Handle, CurrentRow}, nullptr,
374 Name + ".load");
375 }
376
377 Extracts.push_back(Builder.CreateExtractValue(CBufLoad, {CurrentIndex++},
378 Name + ".extract"));
379 }
380
381 // Finally, we build up the original loaded value.
382 Value *Result = PoisonValue::get(Ty);
383 for (int I = 0, E = Extracts.size(); I < E; ++I)
384 Result = Builder.CreateInsertElement(
385 Result, Extracts[I], Builder.getInt32(I), Name + formatv(".upto{}", I));
386 LI->replaceAllUsesWith(Result);
387}

389static void createLoadIntrinsic(IntrinsicInst *II, LoadInst *LI,
390 dxil::ResourceTypeInfo &RTI) {
391 switch (RTI.getResourceKind()) {
392 case dxil::ResourceKind::TypedBuffer:
393 return createTypedBufferLoad(II, LI, RTI);
394 case dxil::ResourceKind::RawBuffer:
395 case dxil::ResourceKind::StructuredBuffer:
396 return createRawLoad(II, LI, RTI);
397 case dxil::ResourceKind::CBuffer:
398 return createCBufferLoad(II, LI, RTI);
399 case dxil::ResourceKind::Texture1D:
400 case dxil::ResourceKind::Texture2D:
401 case dxil::ResourceKind::Texture2DMS:
402 case dxil::ResourceKind::Texture3D:
403 case dxil::ResourceKind::TextureCube:
404 case dxil::ResourceKind::Texture1DArray:
405 case dxil::ResourceKind::Texture2DArray:
406 case dxil::ResourceKind::Texture2DMSArray:
407 case dxil::ResourceKind::TextureCubeArray:
408 case dxil::ResourceKind::FeedbackTexture2D:
409 case dxil::ResourceKind::FeedbackTexture2DArray:

414 case dxil::ResourceKind::RTAccelerationStructure:

416 case dxil::ResourceKind::NumEntries:
417 llvm_unreachable("Invalid resource kind for load");
418 }
419 llvm_unreachable("Unhandled case in switch");

422static SmallVector<Instruction *> collectBlockUseDef(Instruction *Start) {
423 SmallPtrSet<Instruction *, 32> Visited;
424 SmallVector<Instruction *, 32> Worklist;
425 SmallVector<Instruction *> Out;
426 auto *BB = Start->getParent();
427
428 // Seed with direct users in this block.
429 for (User *U : Start->users()) {
430 if (auto *I = dyn_cast<Instruction>(U)) {
431 if (I->getParent() == BB)
432 Worklist.push_back(I);
433 }
434 }
435
436 // BFS over transitive users, constrained to the same block.
437 while (!Worklist.empty()) {
438 Instruction *I = Worklist.pop_back_val();
439 if (!Visited.insert(I).second)
440 continue;
441 Out.push_back(I);
442
443 for (User *U : I->users()) {
444 if (auto *J = dyn_cast<Instruction>(U)) {
445 if (J->getParent() == BB)
446 Worklist.push_back(J);
447 }
448 }
449 for (Use &V : I->operands()) {
450 if (auto *J = dyn_cast<Instruction>(V)) {
451 if (J->getParent() == BB && V != Start)
452 Worklist.push_back(J);
453 }
454 }
455 }
456
457 // Order results in program order.
458 DenseMap<const Instruction *, unsigned> Ord;
459 unsigned Idx = 0;
460 for (Instruction &I : *BB)
461 Ord[&I] = Idx++;
462
463 llvm::sort(Out, [&](Instruction *A, Instruction *B) {
464 return Ord.lookup(A) < Ord.lookup(B);
465 });
466
467 return Out;
468}

470static void phiNodeRemapHelper(PHINode *Phi, BasicBlock *BB,
471 IRBuilder<> &Builder,
472 SmallVector<Instruction *> &UsesInBlock) {
473
474 ValueToValueMapTy VMap;
475 Value *Val = Phi->getIncomingValueForBlock(BB);
476 VMap[Phi] = Val;
477 Builder.SetInsertPoint(&BB->back());
478 for (Instruction *I : UsesInBlock) {
479 // don't clone over the Phi just remap them
480 if (auto *PhiNested = dyn_cast<PHINode>(I)) {
481 VMap[PhiNested] = PhiNested->getIncomingValueForBlock(BB);
482 continue;
483 }
484 Instruction *Clone = I->clone();
485 RemapInstruction(Clone, VMap,
486 RF_NoModuleLevelChanges | RF_IgnoreMissingLocals);
487 Builder.Insert(Clone);
488 VMap[I] = Clone;
489 }
490}

492static void phiNodeReplacement(IntrinsicInst *II,
493 SmallVectorImpl<Instruction *> &PrevBBDeadInsts,
494 SetVector<BasicBlock *> &DeadBB) {
495 SmallVector<Instruction *> CurrBBDeadInsts;
496 for (User *U : II->users()) {
497 auto *Phi = dyn_cast<PHINode>(U);
498 if (!Phi)
499 continue;
500
501 IRBuilder<> Builder(Phi);
502 SmallVector<Instruction *> UsesInBlock = collectBlockUseDef(Phi);
503 bool HasReturnUse = isa<ReturnInst>(UsesInBlock.back());
504
505 for (unsigned I = 0, E = Phi->getNumIncomingValues(); I < E; I++) {
506 auto *CurrIncomingBB = Phi->getIncomingBlock(I);
507 phiNodeRemapHelper(Phi, CurrIncomingBB, Builder, UsesInBlock);
508 if (HasReturnUse)
509 PrevBBDeadInsts.push_back(&CurrIncomingBB->back());
510 }
511
512 CurrBBDeadInsts.push_back(Phi);
513
514 for (Instruction *I : UsesInBlock) {
515 CurrBBDeadInsts.push_back(I);
516 }
517 if (HasReturnUse) {
518 BasicBlock *PhiBB = Phi->getParent();
519 DeadBB.insert(PhiBB);
520 }
521 }
522 // Traverse the now-dead instructions in RPO and remove them.
523 for (Instruction *Dead : llvm::reverse(CurrBBDeadInsts))
524 Dead->eraseFromParent();
525 CurrBBDeadInsts.clear();
526}

528static void replaceAccess(IntrinsicInst *II, dxil::ResourceTypeInfo &RTI) {
529 SmallVector<User *> Worklist;
530 for (User *U : II->users())
531 Worklist.push_back(U);
532
533 SmallVector<Instruction *> DeadInsts;
534 while (!Worklist.empty()) {
535 User *U = Worklist.back();
536 Worklist.pop_back();
537
538 if (auto *GEP = dyn_cast<GetElementPtrInst>(U)) {
539 for (User *U : GEP->users())
540 Worklist.push_back(U);
541 DeadInsts.push_back(GEP);
542
543 } else if (auto *SI = dyn_cast<StoreInst>(U)) {
544 assert(SI->getValueOperand() != II && "Pointer escaped!");
545 createStoreIntrinsic(II, SI, RTI);
546 DeadInsts.push_back(SI);
547
548 } else if (auto *LI = dyn_cast<LoadInst>(U)) {
549 createLoadIntrinsic(II, LI, RTI);
550 DeadInsts.push_back(LI);
551 } else
552 llvm_unreachable("Unhandled instruction - pointer escaped?");
553 }
554
555 // Traverse the now-dead instructions in RPO and remove them.
556 for (Instruction *Dead : llvm::reverse(DeadInsts))
557 Dead->eraseFromParent();
558 II->eraseFromParent();
559}

561static bool transformResourcePointers(Function &F, DXILResourceTypeMap &DRTM) {
562 SmallVector<std::pair<IntrinsicInst *, dxil::ResourceTypeInfo>> Resources;
563 SetVector<BasicBlock *> DeadBB;
564 SmallVector<Instruction *> PrevBBDeadInsts;
565 for (BasicBlock &BB : make_early_inc_range(F)) {
566 for (Instruction &I : make_early_inc_range(BB))
567 if (auto *II = dyn_cast<IntrinsicInst>(&I))
568 if (II->getIntrinsicID() == Intrinsic::dx_resource_getpointer)
569 phiNodeReplacement(II, PrevBBDeadInsts, DeadBB);
570
571 for (Instruction &I : BB)
572 if (auto *II = dyn_cast<IntrinsicInst>(&I))
573 if (II->getIntrinsicID() == Intrinsic::dx_resource_getpointer) {
574 auto *HandleTy = cast<TargetExtType>(II->getArgOperand(0)->getType());
575 Resources.emplace_back(II, DRTM[HandleTy]);
576 }
577 }
578 for (auto *Dead : PrevBBDeadInsts)
579 Dead->eraseFromParent();
580 PrevBBDeadInsts.clear();
581 for (auto *Dead : DeadBB)
582 Dead->eraseFromParent();
583 DeadBB.clear();
584
585 for (auto &[II, RI] : Resources)
586 replaceAccess(II, RI);
587
588 return !Resources.empty();
589}

591PreservedAnalyses DXILResourceAccess::run(Function &F,
592 FunctionAnalysisManager &FAM) {
593 auto &MAMProxy = FAM.getResult<ModuleAnalysisManagerFunctionProxy>(F);
594 DXILResourceTypeMap *DRTM =
595 MAMProxy.getCachedResult<DXILResourceTypeAnalysis>(*F.getParent());
596 assert(DRTM && "DXILResourceTypeAnalysis must be available");
597
598 bool MadeChanges = transformResourcePointers(F, *DRTM);
599 if (!MadeChanges)
600 return PreservedAnalyses::all();
601
602 PreservedAnalyses PA;
603 PA.preserve<DXILResourceTypeAnalysis>();
604 PA.preserve<DominatorTreeAnalysis>();
605 return PA;
606}
