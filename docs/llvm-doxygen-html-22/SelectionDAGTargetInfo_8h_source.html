63 /// Emit target-specific code that performs a memcpy.
64 /// This can be used by targets to provide code sequences for cases
65 /// that don't fit the target's parameters for simple loads/stores and can be
66 /// more efficient than using a library call. This function can return a null
67 /// SDValue if the target declines to use custom code and a different
68 /// lowering strategy should be used.
69 ///
70 /// If AlwaysInline is true, the size is constant and the target should not
71 /// emit any calls and is strongly encouraged to attempt to emit inline code
72 /// even if it is beyond the usual threshold because this intrinsic is being
73 /// expanded in a place where calls are not feasible (e.g. within the prologue
74 /// for another call). If the target chooses to decline an AlwaysInline
75 /// request here, legalize will resort to using simple loads and stores.
76 virtual SDValue EmitTargetCodeForMemcpy(SelectionDAG &DAG, const SDLoc &dl,

81 MachinePointerInfo DstPtrInfo,
82 MachinePointerInfo SrcPtrInfo) const {
83 return SDValue();

189/// Proxy class that targets should inherit from if they wish to use
190/// the generated node descriptions.
191class SelectionDAGGenTargetInfo : public SelectionDAGTargetInfo {
192protected:
193 const SDNodeInfo &GenNodeInfo;
194
195 explicit SelectionDAGGenTargetInfo(const SDNodeInfo &GenNodeInfo)
196 : GenNodeInfo(GenNodeInfo) {}
197
198public:
199 ~SelectionDAGGenTargetInfo() override;
200
201 const char *getTargetNodeName(unsigned Opcode) const override {
202 assert(GenNodeInfo.hasDesc(Opcode) &&
203 "The name should be provided by the derived class");
204 return GenNodeInfo.getName(Opcode).data();
205 }
206
207 bool isTargetMemoryOpcode(unsigned Opcode) const override {
208 if (GenNodeInfo.hasDesc(Opcode))
209 return GenNodeInfo.getDesc(Opcode).hasProperty(SDNPMemOperand);
210 return false;
211 }
212
213 bool isTargetStrictFPOpcode(unsigned Opcode) const override {
214 if (GenNodeInfo.hasDesc(Opcode))
215 return GenNodeInfo.getDesc(Opcode).hasFlag(SDNFIsStrictFP);
216 return false;
217 }
218
219 void verifyTargetNode(const SelectionDAG &DAG,
220 const SDNode *N) const override {
221 if (GenNodeInfo.hasDesc(N->getOpcode()))
222 GenNodeInfo.verifyNode(DAG, N);
223 }
224};
