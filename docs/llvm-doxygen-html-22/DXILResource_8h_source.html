40/// The dx.RawBuffer target extension type
41///
42/// `target("dx.RawBuffer", Type, IsWriteable, IsROV)`
43class RawBufferExtType : public TargetExtType {
44public:
45 RawBufferExtType() = delete;
46 RawBufferExtType(const RawBufferExtType &) = delete;
47 RawBufferExtType &operator=(const RawBufferExtType &) = delete;
48
49 bool isStructured() const {
50 // TODO: We need to be more prescriptive here, but since there's some debate
51 // over whether byte address buffer should have a void type or an i8 type,
52 // accept either for now.
53 Type *Ty = getTypeParameter(0);
54 return !Ty->isVoidTy() && !Ty->isIntegerTy(8);
55 }
56
57 Type *getResourceType() const {
58 return isStructured() ? getTypeParameter(0) : nullptr;
59 }
60 bool isWriteable() const { return getIntParameter(0); }
61 bool isROV() const { return getIntParameter(1); }
62
63 static bool classof(const TargetExtType *T) {
64 return T->getName() == "dx.RawBuffer";
65 }
66 static bool classof(const Type *T) {
67 return isa<TargetExtType>(T) && classof(cast<TargetExtType>(T));
68 }

371class ResourceInfo {
372public:
373 struct ResourceBinding {
374 uint32_t RecordID;
375 uint32_t Space;
376 uint32_t LowerBound;
377 uint32_t Size;
378
379 bool operator==(const ResourceBinding &RHS) const {
380 return std::tie(RecordID, Space, LowerBound, Size) ==
381 std::tie(RHS.RecordID, RHS.Space, RHS.LowerBound, RHS.Size);
382 }
383 bool operator!=(const ResourceBinding &RHS) const {
384 return !(*this == RHS);
385 }
386 bool operator<(const ResourceBinding &RHS) const {
387 return std::tie(RecordID, Space, LowerBound, Size) <
388 std::tie(RHS.RecordID, RHS.Space, RHS.LowerBound, RHS.Size);
389 }
390 bool overlapsWith(const ResourceBinding &RHS) const {
391 if (Space != RHS.Space)
392 return false;
393 if (Size == UINT32_MAX)
394 return LowerBound < RHS.LowerBound;
395 return LowerBound + Size - 1 >= RHS.LowerBound;
396 }
397 };
398
399private:
400 ResourceBinding Binding;
401 TargetExtType *HandleTy;
402 StringRef Name;
403 GlobalVariable *Symbol = nullptr;
404
405public:
406 bool GloballyCoherent = false;
407 ResourceCounterDirection CounterDirection = ResourceCounterDirection::Unknown;
408
409 ResourceInfo(uint32_t RecordID, uint32_t Space, uint32_t LowerBound,
410 uint32_t Size, TargetExtType *HandleTy, StringRef Name = "",
411 GlobalVariable *Symbol = nullptr)
412 : Binding{RecordID, Space, LowerBound, Size}, HandleTy(HandleTy),
413 Name(Name), Symbol(Symbol) {}
414
415 void setBindingID(unsigned ID) { Binding.RecordID = ID; }
416
417 bool hasCounter() const {
418 return CounterDirection != ResourceCounterDirection::Unknown;
419 }
420
421 const ResourceBinding &getBinding() const { return Binding; }
422 TargetExtType *getHandleTy() const { return HandleTy; }
423 StringRef getName() const { return Name; }
424
425 bool hasSymbol() const { return Symbol; }
426 LLVM_ABI GlobalVariable *createSymbol(Module &M, StructType *Ty);
427 LLVM_ABI MDTuple *getAsMetadata(Module &M, dxil::ResourceTypeInfo &RTI) const;
428
429 LLVM_ABI std::pair<uint32_t, uint32_t>
430 getAnnotateProps(Module &M, dxil::ResourceTypeInfo &RTI) const;
431
432 bool operator==(const ResourceInfo &RHS) const {
433 return std::tie(Binding, HandleTy, Symbol, Name) ==
434 std::tie(RHS.Binding, RHS.HandleTy, RHS.Symbol, RHS.Name);
435 }
436 bool operator!=(const ResourceInfo &RHS) const { return !(*this == RHS); }
437 bool operator<(const ResourceInfo &RHS) const {
438 return Binding < RHS.Binding;
439 }
440
441 LLVM_ABI void print(raw_ostream &OS, dxil::ResourceTypeInfo &RTI,
442 const DataLayout &DL) const;
443};

656// DXILResourceBindingInfo stores the results of DXILResourceBindingAnalysis
657// which analyses all llvm.dx.resource.handlefrombinding calls in the module
658// and puts together lists of used virtual register spaces and available
659// virtual register slot ranges for each binding type.
660// It also stores additional information found during the analysis such as
661// whether the module uses implicit bindings or if any of the bindings overlap.
662//
663// This information will be used in DXILResourceImplicitBindings pass to assign
664// register slots to resources with implicit bindings, and in a
665// post-optimization validation pass that will raise diagnostic about
666// overlapping bindings.
667class DXILResourceBindingInfo {
668 hlsl::BindingInfo Bindings;
669 bool HasImplicitBinding = false;
670 bool HasOverlappingBinding = false;
671
672 // Populate the resource binding info given explicit resource binding calls
673 // in the module.
674 void populate(Module &M, DXILResourceTypeMap &DRTM);
675
676public:
677 bool hasImplicitBinding() const { return HasImplicitBinding; }
