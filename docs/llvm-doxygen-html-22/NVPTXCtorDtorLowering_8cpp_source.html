59static Function *createInitOrFiniKernelFunction(Module &M, bool IsCtor) {
60 StringRef InitOrFiniKernelName =
61 IsCtor ? "nvptx$device$init" : "nvptx$device$fini";
62 if (M.getFunction(InitOrFiniKernelName))
63 return nullptr;
64
65 Function *InitOrFiniKernel = Function::createWithDefaultAttr(
66 FunctionType::get(Type::getVoidTy(M.getContext()), false),
67 GlobalValue::WeakODRLinkage, 0, InitOrFiniKernelName, &M);
68 addKernelAttrs(InitOrFiniKernel);
69
70 return InitOrFiniKernel;
71}

73// We create the IR required to call each callback in this section. This is
74// equivalent to the following code. Normally, the linker would provide us with
75// the definitions of the init and fini array sections. The 'nvlink' linker does
76// not do this so initializing these values is done by the runtime.
77//
78// extern "C" void **__init_array_start = nullptr;
79// extern "C" void **__init_array_end = nullptr;
80// extern "C" void **__fini_array_start = nullptr;
81// extern "C" void **__fini_array_end = nullptr;
82//
83// using InitCallback = void();
84// using FiniCallback = void();
85//
86// void call_init_array_callbacks() {
87// for (auto start = __init_array_start; start != __init_array_end; ++start)
88// reinterpret_cast<InitCallback *>(*start)();
89// }
90//
91// void call_fini_array_callbacks() {
92// size_t fini_array_size = __fini_array_end - __fini_array_start;
93// for (size_t i = fini_array_size; i > 0; --i)
94// reinterpret_cast<FiniCallback *>(__fini_array_start[i - 1])();
95// }
96static void createInitOrFiniCalls(Function &F, bool IsCtor) {
97 Module &M = *F.getParent();
98 LLVMContext &C = M.getContext();
99
100 IRBuilder<> IRB(BasicBlock::Create(C, "entry", &F));
101 auto *LoopBB = BasicBlock::Create(C, "while.entry", &F);
102 auto *ExitBB = BasicBlock::Create(C, "while.end", &F);
103 Type *PtrTy = IRB.getPtrTy(llvm::ADDRESS_SPACE_GLOBAL);
104
105 auto *Begin = M.getOrInsertGlobal(
106 IsCtor ? "__init_array_start" : "__fini_array_start",
107 PointerType::get(C, 0), [&]() {
108 auto *GV = new GlobalVariable(
109 M, PointerType::get(C, 0),
110 /*isConstant=*/false, GlobalValue::WeakAnyLinkage,
111 Constant::getNullValue(PointerType::get(C, 0)),
112 IsCtor ? "__init_array_start" : "__fini_array_start",
113 /*InsertBefore=*/nullptr, GlobalVariable::NotThreadLocal,
114 /*AddressSpace=*/llvm::ADDRESS_SPACE_GLOBAL);
115 GV->setVisibility(GlobalVariable::ProtectedVisibility);
116 return GV;
117 });
118 auto *End = M.getOrInsertGlobal(
119 IsCtor ? "__init_array_end" : "__fini_array_end", PointerType::get(C, 0),
120 [&]() {
121 auto *GV = new GlobalVariable(
122 M, PointerType::get(C, 0),
123 /*isConstant=*/false, GlobalValue::WeakAnyLinkage,
124 Constant::getNullValue(PointerType::get(C, 0)),
125 IsCtor ? "__init_array_end" : "__fini_array_end",
126 /*InsertBefore=*/nullptr, GlobalVariable::NotThreadLocal,
127 /*AddressSpace=*/llvm::ADDRESS_SPACE_GLOBAL);
128 GV->setVisibility(GlobalVariable::ProtectedVisibility);
129 return GV;
130 });
131
132 // The constructor type is suppoed to allow using the argument vectors, but
133 // for now we just call them with no arguments.
134 auto *CallBackTy = FunctionType::get(IRB.getVoidTy(), {});
135
136 // The destructor array must be called in reverse order. Get an expression to
137 // the end of the array and iterate backwards in that case.
138 Value *BeginVal = IRB.CreateLoad(Begin->getType(), Begin, "begin");
139 Value *EndVal = IRB.CreateLoad(Begin->getType(), End, "stop");
140 if (!IsCtor) {
141 auto *BeginInt = IRB.CreatePtrToInt(BeginVal, IntegerType::getInt64Ty(C));
142 auto *EndInt = IRB.CreatePtrToInt(EndVal, IntegerType::getInt64Ty(C));
143 auto *SubInst = IRB.CreateSub(EndInt, BeginInt);
144 auto *Offset = IRB.CreateAShr(
145 SubInst, ConstantInt::get(IntegerType::getInt64Ty(C), 3), "offset",
146 /*IsExact=*/true);
147 auto *ValuePtr = IRB.CreateGEP(PointerType::get(C, 0), BeginVal,
148 ArrayRef<Value *>({Offset}));
149 EndVal = BeginVal;
150 BeginVal =
151 IRB.CreateInBoundsGEP(PointerType::get(C, 0), ValuePtr,
152 ArrayRef<Value *>(ConstantInt::getAllOnesValue(
153 IntegerType::getInt64Ty(C))),
154 "start");
155 }
156 IRB.CreateCondBr(
157 IRB.CreateCmp(IsCtor ? ICmpInst::ICMP_NE : ICmpInst::ICMP_UGE, BeginVal,
158 EndVal),
159 LoopBB, ExitBB);
160 IRB.SetInsertPoint(LoopBB);
161 auto *CallBackPHI = IRB.CreatePHI(PtrTy, 2, "ptr");
162 auto *CallBack = IRB.CreateLoad(IRB.getPtrTy(F.getAddressSpace()),
163 CallBackPHI, "callback");
164 IRB.CreateCall(CallBackTy, CallBack);
165 auto *NewCallBack =
166 IRB.CreateConstGEP1_64(PtrTy, CallBackPHI, IsCtor ? 1 : -1, "next");
167 auto *EndCmp = IRB.CreateCmp(IsCtor ? ICmpInst::ICMP_EQ : ICmpInst::ICMP_ULT,
168 NewCallBack, EndVal, "end");
169 CallBackPHI->addIncoming(BeginVal, &F.getEntryBlock());
170 CallBackPHI->addIncoming(NewCallBack, LoopBB);
171 IRB.CreateCondBr(EndCmp, ExitBB, LoopBB);
172 IRB.SetInsertPoint(ExitBB);
173 IRB.CreateRetVoid();
174}

176static bool createInitOrFiniGlobals(Module &M, GlobalVariable *GV,
177 bool IsCtor) {
178 ConstantArray *GA = dyn_cast<ConstantArray>(GV->getInitializer());
179 if (!GA || GA->getNumOperands() == 0)
180 return false;
181
182 // NVPTX has no way to emit variables at specific sections or support for
183 // the traditional constructor sections. Instead, we emit mangled global
184 // names so the runtime can build the list manually.
185 for (Value *V : GA->operands()) {
186 auto *CS = cast<ConstantStruct>(V);
187 auto *F = cast<Constant>(CS->getOperand(1));
188 uint64_t Priority = cast<ConstantInt>(CS->getOperand(0))->getSExtValue();
189 std::string PriorityStr = "." + std::to_string(Priority);
190 // We append a semi-unique hash and the priority to the global name.
191 std::string GlobalID =
192 !GlobalStr.empty() ? GlobalStr : getHash(M.getSourceFileName());
193 std::string NameStr =
194 ((IsCtor ? "__init_array_object_" : "__fini_array_object_") +
195 F->getName() + "_" + GlobalID + "_" + std::to_string(Priority))
196 .str();
197 // PTX does not support exported names with '.' in them.
198 llvm::transform(NameStr, NameStr.begin(),
199 [](char c) { return c == '.' ? '_' : c; });
200
201 auto *GV = new GlobalVariable(M, F->getType(), /*IsConstant=*/true,
202 GlobalValue::ExternalLinkage, F, NameStr,
203 nullptr, GlobalValue::NotThreadLocal,
204 /*AddressSpace=*/4);
205 // This isn't respected by Nvidia, simply put here for clarity.
206 GV->setSection(IsCtor ? ".init_array" + PriorityStr
207 : ".fini_array" + PriorityStr);
208 GV->setVisibility(GlobalVariable::ProtectedVisibility);
209 appendToUsed(M, {GV});
210 }
211
212 return true;
213}

215static bool createInitOrFiniKernel(Module &M, StringRef GlobalName,
216 bool IsCtor) {
217 GlobalVariable *GV = M.getGlobalVariable(GlobalName);
218 if (!GV || !GV->hasInitializer())
219 return false;
220
221 if (!createInitOrFiniGlobals(M, GV, IsCtor))
222 return false;
223
224 if (!CreateKernels)
225 return true;
226
227 Function *InitOrFiniKernel = createInitOrFiniKernelFunction(M, IsCtor);
228 if (!InitOrFiniKernel)
229 return false;
230
231 createInitOrFiniCalls(*InitOrFiniKernel, IsCtor);
232
233 GV->eraseFromParent();
234 return true;
235}

237static bool lowerCtorsAndDtors(Module &M) {
238 bool Modified = false;
239 Modified |= createInitOrFiniKernel(M, "llvm.global_ctors", /*IsCtor =*/true);
240 Modified |= createInitOrFiniKernel(M, "llvm.global_dtors", /*IsCtor =*/false);
241 return Modified;
242}
