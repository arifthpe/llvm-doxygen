37static MCAsmInfo *createVEMCAsmInfo(const MCRegisterInfo &MRI, const Triple &TT,
38 const MCTargetOptions &Options) {
39 MCAsmInfo *MAI = new VEELFMCAsmInfo(TT);
40 unsigned Reg = MRI.getDwarfRegNum(VE::SX11, true);
41 MCCFIInstruction Inst = MCCFIInstruction::cfiDefCfa(nullptr, Reg, 0);
42 MAI->addInitialFrameState(Inst);
43 return MAI;
44}

88extern "C" LLVM_ABI LLVM_EXTERNAL_VISIBILITY void LLVMInitializeVETargetMC() {
89 // Register the MC asm info.
90 RegisterMCAsmInfoFn X(getTheVETarget(), createVEMCAsmInfo);
91
92 for (Target *T : {&getTheVETarget()}) {
93 // Register the MC instruction info.
94 TargetRegistry::RegisterMCInstrInfo(*T, createVEMCInstrInfo);
95
96 // Register the MC register info.
97 TargetRegistry::RegisterMCRegInfo(*T, createVEMCRegisterInfo);
98
99 // Register the MC subtarget info.
100 TargetRegistry::RegisterMCSubtargetInfo(*T, createVEMCSubtargetInfo);
101
102 // Register the MC Code Emitter.
103 TargetRegistry::RegisterMCCodeEmitter(*T, createVEMCCodeEmitter);
104
105 // Register the asm backend.
106 TargetRegistry::RegisterMCAsmBackend(*T, createVEAsmBackend);
107
108 // Register the object target streamer.
109 TargetRegistry::RegisterObjectTargetStreamer(*T,
110 createObjectTargetStreamer);
111
112 // Register the asm streamer.
113 TargetRegistry::RegisterAsmTargetStreamer(*T, createTargetAsmStreamer);
114
115 // Register the null streamer.
116 TargetRegistry::RegisterNullTargetStreamer(*T, createNullTargetStreamer);
117
118 // Register the MCInstPrinter
119 TargetRegistry::RegisterMCInstPrinter(*T, createVEMCInstPrinter);
120 }
121}
