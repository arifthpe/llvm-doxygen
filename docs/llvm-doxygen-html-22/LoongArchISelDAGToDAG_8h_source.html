69 // Return the LoongArch branch opcode that matches the given DAG integer
70 // condition code. The CondCode must be one of those supported by the
71 // LoongArch ISA (see translateSetCCForBranch).
72 static unsigned getBranchOpcForIntCC(ISD::CondCode CC) {
73 switch (CC) {
74 default:
75 llvm_unreachable("Unsupported CondCode");
76 case ISD::SETEQ:
77 return LoongArch::BEQ;
78 case ISD::SETNE:
79 return LoongArch::BNE;
80 case ISD::SETLT:
81 return LoongArch::BLT;
82 case ISD::SETGE:
83 return LoongArch::BGE;
84 case ISD::SETULT:
85 return LoongArch::BLTU;
86 case ISD::SETUGE:
87 return LoongArch::BGEU;
88 }
89 }
