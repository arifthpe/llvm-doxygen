155bool DirectXTargetMachine::addPassesToEmitFile(
156 PassManagerBase &PM, raw_pwrite_stream &Out, raw_pwrite_stream *DwoOut,
157 CodeGenFileType FileType, bool DisableVerify,
158 MachineModuleInfoWrapperPass *MMIWP) {
159 TargetPassConfig *PassConfig = createPassConfig(PM);
160 PassConfig->addCodeGenPrepare();
161
162 switch (FileType) {
163 case CodeGenFileType::AssemblyFile:
164 PM.add(createDXILPrettyPrinterLegacyPass(Out));
165 PM.add(createPrintModulePass(Out, "", true));
166 break;
167 case CodeGenFileType::ObjectFile:
168 if (TargetPassConfig::willCompleteCodeGenPipeline()) {
169 PM.add(createDXILEmbedderPass());
170 // We embed the other DXContainer globals after embedding DXIL so that the
171 // globals don't pollute the DXIL.
172 PM.add(createDXContainerGlobalsPass());
173
174 if (!MMIWP)
175 MMIWP = new MachineModuleInfoWrapperPass(this);
176 PM.add(MMIWP);
177 if (addAsmPrinter(PM, Out, DwoOut, FileType,
178 MMIWP->getMMI().getContext()))
179 return true;
180 } else
181 PM.add(createDXILWriterPass(Out));
182 break;
183 case CodeGenFileType::Null:
184 break;
185 }
186 return false;
187}
